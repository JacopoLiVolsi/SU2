547  unsigned long iPoint;

565

568  GridVel = ( Disp_nP1[iDim] - Disp_n[iDim] ) / TimeStep;

570  GridVel = ( 3.0*Disp_nP1[iDim] - 4.0*Disp_n[iDim]

646 void CMeshSolver::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter, bool val_update_geo) {
647 
648  /*--- Restart the solution from file information ---*/
649  unsigned short iDim;
650  unsigned long index;
651 
652  su2double curr_coord, displ;
653 
654  string UnstExt, text_line;
655  ifstream restart_file;
656 
657  string restart_filename = config->GetSolution_FileName();
658 
659  int counter = 0;
660  long iPoint_Local = 0; unsigned long iPoint_Global = 0;
661  unsigned long iPoint_Global_Local = 0;
662  unsigned short rbuf_NotMatching = 0, sbuf_NotMatching = 0;
663 
664  restart_filename = config->GetFilename(restart_filename, "", val_iter);
665 
666  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
667 
668  if (config->GetRead_Binary_Restart()) {
669  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
670  } else {
671  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
672  }
673 
674  /*--- Load data from the restart into correct containers. ---*/
675  counter = 0;
676  for (iPoint_Global = 0; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++ ) {
677 
678  /*--- Retrieve local index. If this node from the restart file lives
679  on the current processor, we will load and instantiate the vars. ---*/
680 
681  iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
682 
683  if (iPoint_Local > -1) {
684 
685  /*--- We need to store this point's data, so jump to the correct
686  offset in the buffer of data from the restart file and load it. ---*/
687 
688  index = counter*Restart_Vars[1];
689  for (iDim = 0; iDim < nDim; iDim++){
690  /*--- Update the coordinates of the mesh ---*/
691  curr_coord = Restart_Data[index+iDim];
695 
696  /*--- Store the displacements computed as the current coordinates
697  minus the coordinates of the reference mesh file ---*/
698  displ = curr_coord - nodes->GetMesh_Coord(iPoint_Local, iDim);
699  nodes->SetSolution(iPoint_Local, iDim, displ);
700  }
701  iPoint_Global_Local++;
702 
703  /*--- Increment the overall counter for how many points have been loaded. ---*/
704  counter++;
705  }
706 
707  }
708 
709  /*--- Detect a wrong solution file ---*/
710 
711  if (iPoint_Global_Local < nPointDomain) { sbuf_NotMatching = 1; }
712 
713  SU2_MPI::Allreduce(&sbuf_NotMatching, &rbuf_NotMatching, 1, MPI_UNSIGNED_SHORT, MPI_SUM, MPI_COMM_WORLD);
714 
715  if (rbuf_NotMatching != 0) {
716  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" doesn't match with the mesh file!\n") +
717  string("It could be empty lines at the end of the file."), CURRENT_FUNCTION);
718  }
719 
720  /*--- Communicate the loaded displacements. ---*/
721  solver[MESH_0][MESH_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
722  solver[MESH_0][MESH_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
723 
724  /*--- Communicate the new coordinates at the halos ---*/
725  geometry[MESH_0]->InitiateComms(geometry[MESH_0], config, COORDINATES);
726  geometry[MESH_0]->CompleteComms(geometry[MESH_0], config, COORDINATES);
727 
728  /*--- Recompute the edges and dual mesh control volumes in the
729  domain and on the boundaries. ---*/
730  UpdateDualGrid(geometry[MESH_0], config);
731 
732  /*--- For time-domain problems, we need to compute the grid velocities ---*/
733  if (time_domain){
734  /*--- Update the old geometry (coordinates n and n-1) ---*/
735  Restart_OldGeometry(geometry[MESH_0], config);
736  /*--- Once Displacement_n and Displacement_n1 are filled,
737  we can compute the Grid Velocity ---*/
738  ComputeGridVelocity(geometry[MESH_0], config);
739  }
740 
741  /*--- Update the multigrid structure after setting up the finest grid,
742  including computing the grid velocities on the coarser levels
743  when the problem is unsteady. ---*/
744  UpdateMultiGrid(geometry, config);
745 
746  /*--- Store the boundary displacements at the Bound_Disp variable. ---*/
747 
748  unsigned short iMarker;
749  unsigned long iVertex, iNode;
750 
751  su2double VarCoord[3] = {0.0, 0.0, 0.0};
752 
753  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
754 
755  if (config->GetMarker_All_Deform_Mesh(iMarker) == YES) {
756 
757  for (iVertex = 0; iVertex < geometry[MESH_0]->nVertex[iMarker]; iVertex++) {
758 
759  /*--- Get node index ---*/
760  iNode = geometry[MESH_0]->vertex[iMarker][iVertex]->GetNode();
761 
762  /*--- Store it into the current displacement. ---*/
763  for (iDim = 0; iDim < nDim; iDim++){
764  VarCoord[iDim] = nodes->GetSolution(iNode,iDim);
765  }
766 
767  nodes->SetBound_Disp(iNode,VarCoord);
768  }
769  }
770  }
771 
772  /*--- Delete the class memory that is used to load the restart. ---*/
773 
774  if (Restart_Vars != NULL) { delete [] Restart_Vars; Restart_Vars = NULL; }
775  if (Restart_Data != NULL) { delete [] Restart_Data; Restart_Data = NULL; }
776 
777 }
