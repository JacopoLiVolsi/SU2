430  calls it knows the consequence to the derivative information (lost) ---*/
431 
432  /*--- check if self-assignment, otherwise perform deep copy ---*/
433  if ((const void*)this == (const void*)&other) return;
434 
435  /*--- determine if (re-)allocation is needed ---*/

437  delete [] vec_val;
438  vec_val = NULL;

440

446  nVar = other.GetNVar();
447

449  vec_val = new ScalarType[nElm];

460 ScalarType dotProd(const CSysVector<ScalarType> & u, const CSysVector<ScalarType> & v) {
461 
462  /*--- check for consistent sizes ---*/
463  if (u.nElm != v.nElm) {
464  SU2_MPI::Error("Sizes do not match", CURRENT_FUNCTION);
465  }
466 
467  /*--- find local inner product and, if a parallel run, sum over all
468  processors (we use nElemDomain instead of nElem) ---*/
469  ScalarType loc_prod = 0.0;
470  for (unsigned long i = 0; i < u.nElmDomain; i++)
471  loc_prod += u.vec_val[i]*v.vec_val[i];

474 #ifdef HAVE_MPI

477  prod = loc_prod;
478 #endif
479 
480  return prod;
