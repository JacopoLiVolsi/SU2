734 void CFluidIteration::Solve(COutput *output,
735  CIntegration ****integration,
736  CGeometry ****geometry,
737  CSolver *****solver,
738  CNumerics ******numerics,
739  CConfig **config,
740  CSurfaceMovement **surface_movement,
741  CVolumetricMovement ***grid_movement,
742  CFreeFormDefBox*** FFDBox,
743  unsigned short val_iZone,
744  unsigned short val_iInst) {
745 
746  /*--- Boolean to determine if we are running a static or dynamic case ---*/
747  bool steady = !config[val_iZone]->GetTime_Domain();
748 
749  unsigned long Inner_Iter, nInner_Iter = config[val_iZone]->GetnInner_Iter();
750  bool StopCalc = false;
751 
752  /*--- Synchronization point before a single solver iteration.
753  Compute the wall clock time required. ---*/
754 
755 #ifndef HAVE_MPI
756  StartTime = su2double(clock())/su2double(CLOCKS_PER_SEC);
757 #else
758  StartTime = MPI_Wtime();
759 #endif
760 
761  /*--- Preprocess the solver ---*/
762  Preprocess(output, integration, geometry, solver, numerics, config,
763  surface_movement, grid_movement, FFDBox, val_iZone, INST_0);
764 
765  /*--- For steady-state flow simulations, we need to loop over ExtIter for the number of time steps ---*/
766  /*--- However, ExtIter is the number of FSI iterations, so nIntIter is used in this case ---*/
767 
768  for (Inner_Iter = 0; Inner_Iter < nInner_Iter; Inner_Iter++){
769 
770  config[val_iZone]->SetInnerIter(Inner_Iter);
771 
772  /*--- Run a single iteration of the solver ---*/
773  Iterate(output, integration, geometry, solver, numerics, config,
774  surface_movement, grid_movement, FFDBox, val_iZone, INST_0);
775 
776  /*--- Monitor the pseudo-time ---*/
777  StopCalc = Monitor(output, integration, geometry, solver, numerics, config,
778  surface_movement, grid_movement, FFDBox, val_iZone, INST_0);
779 
780  /*--- Output files at intermediate iterations if the problem is single zone ---*/
781 
782  if (singlezone && steady) {
783  Output(output, geometry, solver, config, Inner_Iter, StopCalc, val_iZone, val_iInst);
784  }
785 
786  /*--- If the iteration has converged, break the loop ---*/
787  if (StopCalc) break;
788  }
789 
790  if (multizone && steady) {
791  Output(output, geometry, solver, config,
792  config[val_iZone]->GetOuterIter(),
793  StopCalc, val_iZone, val_iInst);
794 
795  /*--- Set the fluid convergence to false (to make sure outer subiterations converge) ---*/
796 
797  integration[val_iZone][INST_0][FLOW_SOL]->SetConvergence(false);
798  }

801 void CFluidIteration::SetWind_GustField(CConfig *config, CGeometry **geometry, CSolver ***solver) {
802  // The gust is imposed on the flow field via the grid velocities. This method called the Field Velocity Method is described in the
803  // NASA TM2012-217771 - Development, Verification and Use of Gust Modeling in the NASA Computational Fluid Dynamics Code FUN3D
804  // the desired gust is prescribed as the negative of the grid velocity.
805 
806  // If a source term is included to account for the gust field, the method is described by Jones et al. as the Split Velocity Method in
807  // Simulation of Airfoil Gust Responses Using Prescribed Velocities.
808  // In this routine the gust derivatives needed for the source term are calculated when applicable.
809  // If the gust derivatives are zero the source term is also zero.
810  // The source term itself is implemented in the class CSourceWindGust
811 
812  if (rank == MASTER_NODE)
813  cout << endl << "Running simulation with a Wind Gust." << endl;
814  unsigned short iDim, nDim = geometry[MESH_0]->GetnDim(); //We assume nDim = 2
815  if (nDim != 2) {
816  if (rank == MASTER_NODE) {
817  cout << endl << "WARNING - Wind Gust capability is only verified for 2 dimensional simulations." << endl;
818  }
819  }
820 
821  /*--- Gust Parameters from config ---*/
822  unsigned short Gust_Type = config->GetGust_Type();
823  su2double xbegin = config->GetGust_Begin_Loc(); // Location at which the gust begins.
824  su2double L = config->GetGust_WaveLength(); // Gust size
825  su2double tbegin = config->GetGust_Begin_Time(); // Physical time at which the gust begins.
826  su2double gust_amp = config->GetGust_Ampl(); // Gust amplitude
827  su2double n = config->GetGust_Periods(); // Number of gust periods
828  unsigned short GustDir = config->GetGust_Dir(); // Gust direction
829 
830  /*--- Variables needed to compute the gust ---*/
831  unsigned short Kind_Grid_Movement = config->GetKind_GridMovement();
832  unsigned long iPoint;
833  unsigned short iMGlevel, nMGlevel = config->GetnMGLevels();
834 
835  su2double x, y, x_gust, dgust_dx, dgust_dy, dgust_dt;
836  su2double *Gust, *GridVel, *NewGridVel, *GustDer;
837 
838  su2double Physical_dt = config->GetDelta_UnstTime();
839  unsigned long TimeIter = config->GetTimeIter();
840  su2double Physical_t = TimeIter*Physical_dt;
841 
842  su2double Uinf = solver[MESH_0][FLOW_SOL]->GetVelocity_Inf(0); // Assumption gust moves at infinity velocity
843 
844  Gust = new su2double [nDim];
845  NewGridVel = new su2double [nDim];
846  for (iDim = 0; iDim < nDim; iDim++) {
847  Gust[iDim] = 0.0;
848  NewGridVel[iDim] = 0.0;
849  }
850 
851  GustDer = new su2double [3];
852  for (unsigned short i = 0; i < 3; i++) {
853  GustDer[i] = 0.0;
854  }
855 
856  // Vortex variables
857  unsigned long nVortex = 0;
858  vector<su2double> x0, y0, vort_strenth, r_core; //vortex is positive in clockwise direction.
859  if (Gust_Type == VORTEX) {
860  InitializeVortexDistribution(nVortex, x0, y0, vort_strenth, r_core);
861  }
862 
863  /*--- Check to make sure gust lenght is not zero or negative (vortex gust doesn't use this). ---*/
864  if (L <= 0.0 && Gust_Type != VORTEX) {
865  SU2_MPI::Error("The gust length needs to be positive", CURRENT_FUNCTION);
866  }
867 
868  /*--- Loop over all multigrid levels ---*/
869 
870  for (iMGlevel = 0; iMGlevel <= nMGlevel; iMGlevel++) {
871 
872  /*--- Loop over each node in the volume mesh ---*/
873 
874  for (iPoint = 0; iPoint < geometry[iMGlevel]->GetnPoint(); iPoint++) {
875 
876  /*--- Reset the Grid Velocity to zero if there is no grid movement ---*/
877  if (Kind_Grid_Movement == GUST) {
878  for (iDim = 0; iDim < nDim; iDim++)
879  geometry[iMGlevel]->node[iPoint]->SetGridVel(iDim, 0.0);
880  }
881 
882  /*--- initialize the gust and derivatives to zero everywhere ---*/
883 
884  for (iDim = 0; iDim < nDim; iDim++) {Gust[iDim]=0.0;}
885  dgust_dx = 0.0; dgust_dy = 0.0; dgust_dt = 0.0;
886 
887  /*--- Begin applying the gust ---*/
888 
889  if (Physical_t >= tbegin) {
890 
891  x = geometry[iMGlevel]->node[iPoint]->GetCoord()[0]; // x-location of the node.
892  y = geometry[iMGlevel]->node[iPoint]->GetCoord()[1]; // y-location of the node.
893 
894  // Gust coordinate
895  x_gust = (x - xbegin - Uinf*(Physical_t-tbegin))/L;
896 
897  /*--- Calculate the specified gust ---*/
898  switch (Gust_Type) {
899 
900  case TOP_HAT:
901  // Check if we are in the region where the gust is active
902  if (x_gust > 0 && x_gust < n) {
903  Gust[GustDir] = gust_amp;
904  // Still need to put the gust derivatives. Think about this.
905  }
906  break;
907 
908  case SINE:
909  // Check if we are in the region where the gust is active
910  if (x_gust > 0 && x_gust < n) {
911  Gust[GustDir] = gust_amp*(sin(2*PI_NUMBER*x_gust));
912 
913  // Gust derivatives
914  //dgust_dx = gust_amp*2*PI_NUMBER*(cos(2*PI_NUMBER*x_gust))/L;
915  //dgust_dy = 0;
916  //dgust_dt = gust_amp*2*PI_NUMBER*(cos(2*PI_NUMBER*x_gust))*(-Uinf)/L;
917  }
918  break;
919 
920  case ONE_M_COSINE:
921  // Check if we are in the region where the gust is active
922  if (x_gust > 0 && x_gust < n) {
923  Gust[GustDir] = gust_amp*(1-cos(2*PI_NUMBER*x_gust));
924 
925  // Gust derivatives
926  //dgust_dx = gust_amp*2*PI_NUMBER*(sin(2*PI_NUMBER*x_gust))/L;
927  //dgust_dy = 0;
928  //dgust_dt = gust_amp*2*PI_NUMBER*(sin(2*PI_NUMBER*x_gust))*(-Uinf)/L;
929  }
930  break;
931 
932  case EOG:
933  // Check if we are in the region where the gust is active
934  if (x_gust > 0 && x_gust < n) {
935  Gust[GustDir] = -0.37*gust_amp*sin(3*PI_NUMBER*x_gust)*(1-cos(2*PI_NUMBER*x_gust));
936  }
937  break;
938 
939  case VORTEX:
940 
941  /*--- Use vortex distribution ---*/
942  // Algebraic vortex equation.
943  for (unsigned long i=0; i<nVortex; i++) {
944  su2double r2 = pow(x-(x0[i]+Uinf*(Physical_t-tbegin)), 2) + pow(y-y0[i], 2);
945  su2double r = sqrt(r2);
946  su2double v_theta = vort_strenth[i]/(2*PI_NUMBER) * r/(r2+pow(r_core[i],2));
947  Gust[0] = Gust[0] + v_theta*(y-y0[i])/r;
948  Gust[1] = Gust[1] - v_theta*(x-(x0[i]+Uinf*(Physical_t-tbegin)))/r;
949  }
950  break;
951 
952  case NONE: default:
953 
954  /*--- There is no wind gust specified. ---*/
955  if (rank == MASTER_NODE) {
956  cout << "No wind gust specified." << endl;
957  }
958  break;
959 
960  }
961  }
962 
963  /*--- Set the Wind Gust, Wind Gust Derivatives and the Grid Velocities ---*/
964 
965  GustDer[0] = dgust_dx;
966  GustDer[1] = dgust_dy;
967  GustDer[2] = dgust_dt;
968 
969  solver[iMGlevel][FLOW_SOL]->GetNodes()->SetWindGust(iPoint, Gust);
970  solver[iMGlevel][FLOW_SOL]->GetNodes()->SetWindGustDer(iPoint, GustDer);
971 
972  GridVel = geometry[iMGlevel]->node[iPoint]->GetGridVel();
973 
974  /*--- Store new grid velocity ---*/
975 
976  for (iDim = 0; iDim < nDim; iDim++) {
977  NewGridVel[iDim] = GridVel[iDim] - Gust[iDim];
978  geometry[iMGlevel]->node[iPoint]->SetGridVel(iDim, NewGridVel[iDim]);
979  }
980 
981  }
982  }
983 
984  delete [] Gust;
985  delete [] GustDer;
986  delete [] NewGridVel;
987 
988 }
