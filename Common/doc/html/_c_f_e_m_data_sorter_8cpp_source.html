117 void CFEMDataSorter::SortVolumetricConnectivity(CConfig *config, CGeometry *geometry, unsigned short Elem_Type) {
118 
119  /* Determine the number of nodes for this element type. */
120  unsigned short NODES_PER_ELEMENT = 0;
121  switch (Elem_Type) {
122  case TRIANGLE:
123  NODES_PER_ELEMENT = N_POINTS_TRIANGLE;
124  break;
125  case QUADRILATERAL:
126  NODES_PER_ELEMENT = N_POINTS_QUADRILATERAL;
127  break;
128  case TETRAHEDRON:
129  NODES_PER_ELEMENT = N_POINTS_TETRAHEDRON;
130  break;
131  case HEXAHEDRON:
132  NODES_PER_ELEMENT = N_POINTS_HEXAHEDRON;
133  break;
134  case PRISM:
135  NODES_PER_ELEMENT = N_POINTS_PRISM;
136  break;
137  case PYRAMID:
138  NODES_PER_ELEMENT = N_POINTS_PYRAMID;
139  break;
140  default:
141  SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
142  }
143 
144  /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
145  geometrical information for the FEM DG solver. ---*/
146  CMeshFEM_DG *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
147 
148  unsigned long nVolElemOwned = DGGeometry->GetNVolElemOwned();
149  CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
150 
151  const CFEMStandardElement *standardElementsSol = DGGeometry->GetStandardElementsSol();
152 
153  /*--- Determine the number of sub-elements on this rank. ---*/
154  unsigned long nSubElem_Local = 0;
155  for(unsigned long i=0; i<nVolElemOwned; ++i) {
156 
157  /* Determine the necessary data from the corresponding standard elem. */
158  const unsigned short ind = volElem[i].indStandardElement;
159  const unsigned short VTK_Type1 = standardElementsSol[ind].GetVTK_Type1();
160  const unsigned short VTK_Type2 = standardElementsSol[ind].GetVTK_Type2();
161 
162  /* Only store the linear sub elements if they are of
163  the current type that we are storing. */
164  if(Elem_Type == VTK_Type1) nSubElem_Local += standardElementsSol[ind].GetNSubElemsType1();
165  if(Elem_Type == VTK_Type2) nSubElem_Local += standardElementsSol[ind].GetNSubElemsType2();
166  }
167 
168  /* Allocate the memory to store the connectivity if the size is
169  larger than zero. */
170  int *Conn_SubElem = NULL;
171  if(nSubElem_Local > 0) Conn_SubElem = new int[nSubElem_Local*NODES_PER_ELEMENT];
172 
173  /*--- Loop again over the local volume elements and store the global
174  connectivities of the sub-elements. Note one is added to the
175  index value, because visualization softwares typically use
176  1-based indexing. ---*/
177  unsigned long kNode = 0;
178  for(unsigned long i=0; i<nVolElemOwned; ++i) {
179 
180  /* Determine the necessary data from the corresponding standard elem. */
181  const unsigned short ind = volElem[i].indStandardElement;
182  const unsigned short VTK_Type1 = standardElementsSol[ind].GetVTK_Type1();
183  const unsigned short VTK_Type2 = standardElementsSol[ind].GetVTK_Type2();
184 
185  /* Check if the first sub-element is of the required type. */
186  if(Elem_Type == VTK_Type1) {
187 
188  /* Get the number of sub-elements and the local connectivity of
189  the sub-elements. */
190  const unsigned short nSubElems = standardElementsSol[ind].GetNSubElemsType1();
191  const unsigned short *connSubElems = standardElementsSol[ind].GetSubConnType1();
192 
193  /* Store the global connectivities. */
194  const unsigned short kk = NODES_PER_ELEMENT*nSubElems;
195  for(unsigned short k=0; k<kk; ++k, ++kNode)
196  Conn_SubElem[kNode] = connSubElems[k] + volElem[i].offsetDOFsSolGlobal + 1;
197  }
198 
199  /* Check if the second sub-element is of the required type. */
200  if(Elem_Type == VTK_Type2) {
201 
202  /* Get the number of sub-elements and the local connectivity of
203  the sub-elements. */
204  const unsigned short nSubElems = standardElementsSol[ind].GetNSubElemsType2();
205  const unsigned short *connSubElems = standardElementsSol[ind].GetSubConnType2();
206 
207  /* Store the global connectivities. */
208  const unsigned short kk = NODES_PER_ELEMENT*nSubElems;
209  for(unsigned short k=0; k<kk; ++k, ++kNode)
210  Conn_SubElem[kNode] = connSubElems[k] + volElem[i].offsetDOFsSolGlobal + 1;
211  }
212  }
213 
214  /*--- Store the particular global element count in the class data,
215  and set the class data pointer to the connectivity array. ---*/
216  switch (Elem_Type) {
217  case TRIANGLE:
218  nParallel_Tria = nSubElem_Local;
219  if (Conn_Tria_Par != NULL) delete [] Conn_Tria_Par;
220  Conn_Tria_Par = Conn_SubElem;
221  break;
222  case QUADRILATERAL:
223  nParallel_Quad = nSubElem_Local;
224  if (Conn_Quad_Par != NULL) delete [] Conn_Quad_Par;
225  Conn_Quad_Par = Conn_SubElem;
226  break;
227  case TETRAHEDRON:
228  nParallel_Tetr = nSubElem_Local;
229  if (Conn_Tetr_Par != NULL) delete [] Conn_Tetr_Par;
230  Conn_Tetr_Par = Conn_SubElem;
231  break;
232  case HEXAHEDRON:
233  nParallel_Hexa = nSubElem_Local;
234  if (Conn_Hexa_Par != NULL) delete [] Conn_Hexa_Par;
235  Conn_Hexa_Par = Conn_SubElem;
236  break;
237  case PRISM:
238  nParallel_Pris = nSubElem_Local;
239  if (Conn_Pris_Par != NULL) delete [] Conn_Pris_Par;
240  Conn_Pris_Par = Conn_SubElem;
241  break;
242  case PYRAMID:
243  nParallel_Pyra = nSubElem_Local;
244  if (Conn_Pyra_Par != NULL) delete [] Conn_Pyra_Par;
245  Conn_Pyra_Par = Conn_SubElem;
246  break;
247  default:
248  SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
249  break;
250  }
