108 void CRectangularMeshReaderFVM::ComputeRectangularVolumeConnectivity() {
109 
110  /* Set the global count of elements based on the grid dimensions. */
111  numberOfGlobalElements = (nNode-1)*(mNode-1);
112 
113  /* Get a partitioner to help with linear partitioning. */
114  CLinearPartitioner pointPartitioner(numberOfGlobalPoints,0);
115 
116  /* Loop over our analytically defined of elements and store only those
117  that contain a node within our linear partition of points. */
118  numberOfLocalElements = 0;
119  vector<unsigned long> connectivity(N_POINTS_HEXAHEDRON,0);
120  unsigned long globalIndex = 0;
121  for (unsigned long jNode = 0; jNode < mNode-1; jNode++) {
122  for (unsigned long iNode = 0; iNode < nNode-1; iNode++) {
123 
124  /* Compute connectivity based on the i,j index. */
125  connectivity[0] = jNode*nNode + iNode;
126  connectivity[1] = jNode*nNode + iNode + 1;
127  connectivity[2] = (jNode + 1)*nNode + (iNode + 1);
128  connectivity[3] = (jNode + 1)*nNode + iNode;
129 
130  /* Check whether any of the points is in our linear partition. */
131  bool isOwned = false;
132  for (unsigned short i = 0; i < N_POINTS_QUADRILATERAL; i++) {
133  if ((int)pointPartitioner.GetRankContainingIndex(connectivity[i]) == rank) {
134  isOwned = true;
135  }
136  }
137 
138  /* If so, we need to store the element locally. */
139  if (isOwned) {
140  localVolumeElementConnectivity.push_back(globalIndex);
141  localVolumeElementConnectivity.push_back(KindElem);
142  for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++) {
143  localVolumeElementConnectivity.push_back(connectivity[i]);
144  }
145  numberOfLocalElements++;
146  }
147  globalIndex++;
148  }
149  }
