129 void CParallelDataSorter::SortOutputData() {
130 
131  int VARS_PER_POINT = GlobalField_Counter;
132 
133 #ifdef HAVE_MPI
134  SU2_MPI::Request *send_req, *recv_req;
135  SU2_MPI::Status status;
136  int ind;
137 #endif
138 
139  /*--- Allocate the memory that we need for receiving the conn
140  values and then cue up the non-blocking receives. Note that
141  we do not include our own rank in the communications. We will
142  directly copy our own data later. ---*/
143 
144 
145  unsigned long *idRecv = new unsigned long[nPoint_Recv[size]]();
146 
147 #ifdef HAVE_MPI
148  /*--- We need double the number of messages to send both the conn.
149  and the global IDs. ---*/
150 
151  send_req = new SU2_MPI::Request[2*nSends];
152  recv_req = new SU2_MPI::Request[2*nRecvs];
153 
154  unsigned long iMessage = 0;
155  for (int ii=0; ii<size; ii++) {
156  if ((ii != rank) && (nPoint_Recv[ii+1] > nPoint_Recv[ii])) {
157  int ll = VARS_PER_POINT*nPoint_Recv[ii];
158  int kk = nPoint_Recv[ii+1] - nPoint_Recv[ii];
159  int count = VARS_PER_POINT*kk;
160  int source = ii;
161  int tag = ii + 1;
162  SU2_MPI::Irecv(&(doubleBuffer[ll]), count, MPI_DOUBLE, source, tag,
163  MPI_COMM_WORLD, &(recv_req[iMessage]));
164  iMessage++;
165  }
166  }
167 
168  /*--- Launch the non-blocking sends of the connectivity. ---*/
169 
170  iMessage = 0;
171  for (int ii=0; ii<size; ii++) {
172  if ((ii != rank) && (nPoint_Send[ii+1] > nPoint_Send[ii])) {
173  int ll = VARS_PER_POINT*nPoint_Send[ii];
174  int kk = nPoint_Send[ii+1] - nPoint_Send[ii];
175  int count = VARS_PER_POINT*kk;
176  int dest = ii;
177  int tag = rank + 1;
178  SU2_MPI::Isend(&(connSend[ll]), count, MPI_DOUBLE, dest, tag,
179  MPI_COMM_WORLD, &(send_req[iMessage]));
180  iMessage++;
181  }
182  }
183 
184  /*--- Repeat the process to communicate the global IDs. ---*/
185 
186  iMessage = 0;
187  for (int ii=0; ii<size; ii++) {
188  if ((ii != rank) && (nPoint_Recv[ii+1] > nPoint_Recv[ii])) {
189  int ll = nPoint_Recv[ii];
190  int kk = nPoint_Recv[ii+1] - nPoint_Recv[ii];
191  int count = kk;
192  int source = ii;
193  int tag = ii + 1;
194  SU2_MPI::Irecv(&(idRecv[ll]), count, MPI_UNSIGNED_LONG, source, tag,
195  MPI_COMM_WORLD, &(recv_req[iMessage+nRecvs]));
196  iMessage++;
197  }
198  }
199 
200  /*--- Launch the non-blocking sends of the global IDs. ---*/
201 
202  iMessage = 0;
203  for (int ii=0; ii<size; ii++) {
204  if ((ii != rank) && (nPoint_Send[ii+1] > nPoint_Send[ii])) {
205  int ll = nPoint_Send[ii];
206  int kk = nPoint_Send[ii+1] - nPoint_Send[ii];
207  int count = kk;
208  int dest = ii;
209  int tag = rank + 1;
210  SU2_MPI::Isend(&(idSend[ll]), count, MPI_UNSIGNED_LONG, dest, tag,
211  MPI_COMM_WORLD, &(send_req[iMessage+nSends]));
212  iMessage++;
213  }
214  }
215 #endif
216 
217  /*--- Copy my own rank's data into the recv buffer directly. ---*/
218 
219  int mm = VARS_PER_POINT*nPoint_Recv[rank];
220  int ll = VARS_PER_POINT*nPoint_Send[rank];
221  int kk = VARS_PER_POINT*nPoint_Send[rank+1];
222 
223  for (int nn=ll; nn<kk; nn++, mm++) doubleBuffer[mm] = connSend[nn];
224 
225  mm = nPoint_Recv[rank];
226  ll = nPoint_Send[rank];
227  kk = nPoint_Send[rank+1];
228 
229  for (int nn=ll; nn<kk; nn++, mm++) idRecv[mm] = idSend[nn];
230 
231  /*--- Wait for the non-blocking sends and recvs to complete. ---*/
232 
233 #ifdef HAVE_MPI
234  int number = 2*nSends;
235  for (int ii = 0; ii < number; ii++)
236  SU2_MPI::Waitany(number, send_req, &ind, &status);
237 
238  number = 2*nRecvs;
239  for (int ii = 0; ii < number; ii++)
240  SU2_MPI::Waitany(number, recv_req, &ind, &status);
241 
242  delete [] send_req;
243  delete [] recv_req;
244 #endif
245 
246  /*--- Note, passiveDoubleBuffer and doubleBuffer point to the same address.
247  * This is the reason why we have to do the following copy/reordering in two steps. ---*/
248 
249  /*--- Step 1: Extract the underlying double value --- */
250 
251  if (!std::is_same<su2double, passivedouble>::value){
252  for (int jj = 0; jj < VARS_PER_POINT*nPoint_Recv[size]; jj++){
253  const passivedouble tmpVal = SU2_TYPE::GetValue(doubleBuffer[jj]);
254  passiveDoubleBuffer[jj] = tmpVal;
255  /*--- For some AD datatypes a call of the destructor is
256  * necessary to properly delete the AD type ---*/
257  doubleBuffer[jj].~su2double();
258  }
259  }
260 
261  /*--- Step 2: Reorder the data in the buffer --- */
262 
263  passivedouble *tmpBuffer = new passivedouble[nPoint_Recv[size]];
264  for (int jj = 0; jj < VARS_PER_POINT; jj++){
265  for (int ii = 0; ii < nPoint_Recv[size]; ii++){
266  tmpBuffer[idRecv[ii]] = passiveDoubleBuffer[ii*VARS_PER_POINT+jj];
267  }
268  for (int ii = 0; ii < nPoint_Recv[size]; ii++){
269  passiveDoubleBuffer[ii*VARS_PER_POINT+jj] = tmpBuffer[ii];
270  }
271  }
272 
273  delete [] tmpBuffer;
274 
275  /*--- Store the total number of local points my rank has for
276  the current section after completing the communications. ---*/
277 
278  nParallel_Poin = nPoint_Recv[size];
279 
280  /*--- Reduce the total number of points we will write in the output files. ---*/
281 
282 #ifndef HAVE_MPI
283  nGlobal_Poin_Par = nParallel_Poin;
284 #else
285  SU2_MPI::Allreduce(&nParallel_Poin, &nGlobal_Poin_Par, 1,
286  MPI_UNSIGNED_LONG, MPI_SUM, MPI_COMM_WORLD);
287 #endif
288 
289  /*--- Free temporary memory from communications ---*/
290 
291  delete [] idRecv;
292 }

294 void CParallelDataSorter::PrepareSendBuffers(std::vector<unsigned long>& globalID){
295 
296  unsigned long iPoint;
297  unsigned short iProcessor;
298 
299  int VARS_PER_POINT = GlobalField_Counter;
300 
301  /*--- We start with the grid nodes distributed across all procs with
302  no particular ordering assumed. We need to loop through our local partition
303  and decide how many nodes we must send to each other rank in order to
304  have all nodes sorted according to a linear partitioning of the grid
305  nodes, i.e., rank 0 holds the first ~ nGlobalPoint()/nProcessors nodes.
306  First, initialize a counter and flag. ---*/
307 
308  for (iPoint = 0; iPoint < nLocalPoint_Sort; iPoint++ ) {
309 
310  iProcessor = linearPartitioner->GetRankContainingIndex(globalID[iPoint]);
311 
312  /*--- If we have not visited this node yet, increment our
313  number of elements that must be sent to a particular proc. ---*/
314 
315  nPoint_Send[iProcessor+1]++;
316  }
317 
318  /*--- Communicate the number of nodes to be sent/recv'd amongst
319  all processors. After this communication, each proc knows how
320  many cells it will receive from each other processor. ---*/
321 
322 #ifdef HAVE_MPI
323  SU2_MPI::Alltoall(&(nPoint_Send[1]), 1, MPI_INT,
324  &(nPoint_Recv[1]), 1, MPI_INT, MPI_COMM_WORLD);
325 #else
326  nPoint_Recv[1] = nPoint_Send[1];
327 #endif
328 
329  /*--- Prepare to send coordinates. First check how many
330  messages we will be sending and receiving. Here we also put
331  the counters into cumulative storage format to make the
332  communications simpler. ---*/
333 
334  nSends = 0; nRecvs = 0;
335 
336  for (int ii = 0; ii < size; ii++) {
337  if ((ii != rank) && (nPoint_Send[ii+1] > 0)) nSends++;
338  if ((ii != rank) && (nPoint_Recv[ii+1] > 0)) nRecvs++;
339 
340  nPoint_Send[ii+1] += nPoint_Send[ii];
341  nPoint_Recv[ii+1] += nPoint_Recv[ii];
342  }
343 
344  /*--- Allocate memory to hold the connectivity that we are
345  sending. ---*/
346 
347  connSend = NULL;
348  connSend = new su2double[VARS_PER_POINT*nPoint_Send[size]]();
349 
350  /*--- Allocate the data buffer to hold the sorted data. We have to make it large enough
351  * to hold passivedoubles and su2doubles ---*/
352  unsigned short maxSize = max(sizeof(passivedouble), sizeof(su2double));
353  dataBuffer = new char[VARS_PER_POINT*nPoint_Recv[size]*maxSize];
354 
355  /*--- doubleBuffer and passiveDouble buffer use the same memory allocated above using the dataBuffer. ---*/
356 
357  doubleBuffer = reinterpret_cast<su2double*>(dataBuffer);
358  passiveDoubleBuffer = reinterpret_cast<passivedouble*>(dataBuffer);
359 
360  /*--- Allocate arrays for sending the global ID. ---*/
361 
362  idSend = new unsigned long[nPoint_Send[size]]();
363 
364  /*--- Create an index variable to keep track of our index
365  positions as we load up the send buffer. ---*/
366 
367  unsigned long *index = new unsigned long[size]();
368  for (int ii=0; ii < size; ii++) index[ii] = VARS_PER_POINT*nPoint_Send[ii];
369 
370  unsigned long *idIndex = new unsigned long[size]();
371  for (int ii=0; ii < size; ii++) idIndex[ii] = nPoint_Send[ii];
372 
373  Index = new unsigned long[nLocalPoint_Sort]();
374 
375  /*--- Loop through our elements and load the elems and their
376  additional data that we will send to the other procs. ---*/
377 
378  for (iPoint = 0; iPoint < nLocalPoint_Sort; iPoint++) {
379 
380  iProcessor = linearPartitioner->GetRankContainingIndex(globalID[iPoint]);
381 
382  /*--- Load the global ID (minus offset) for sorting the
383  points once they all reach the correct processor. ---*/
384 
385  unsigned long nn = idIndex[iProcessor];
386  idSend[nn] = globalID[iPoint] - linearPartitioner->GetFirstIndexOnRank(iProcessor);
387 
388  /*--- Store the index this point has in the send buffer ---*/
389 
390  Index[iPoint] = index[iProcessor];
391 
392  /*--- Increment the index by the message length ---*/
393 
394  index[iProcessor] += VARS_PER_POINT;
395  idIndex[iProcessor]++;
396 
397 
398  }
399 
400  /*--- Free memory after loading up the send buffer. ---*/
401 
402  delete [] index;
403  delete [] idIndex;
404 }
