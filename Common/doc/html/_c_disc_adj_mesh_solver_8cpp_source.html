201 void CDiscAdjMeshSolver::SetSensitivity(CGeometry *geometry, CSolver **solver, CConfig *config) {
202 
203  unsigned long iPoint;
204  unsigned short iDim;
205  su2double Sensitivity, eps;
206  bool time_stepping = (config->GetTime_Marching() != STEADY);
207 
208  /*--- Extract the sensitivities ---*/
209  ExtractAdjoint_Solution(geometry, config);
210 
211  /*--- Extract the adjoint variables: sensitivities of the boundary displacements ---*/
212  ExtractAdjoint_Variables(geometry, config);
213 
214  /*--- Store the sensitivities in the flow adjoint container ---*/
215  for (iPoint = 0; iPoint < nPoint; iPoint++) {
216 
217  for (iDim = 0; iDim < nDim; iDim++) {
218 
219  /*--- The sensitivity was extracted using ExtractAdjoint_Solution ---*/
220  Sensitivity = nodes->GetSolution(iPoint,iDim);
221 
222  /*--- If sharp edge, set the sensitivity to 0 on that region ---*/
223  if (config->GetSens_Remove_Sharp()) {
224  eps = config->GetVenkat_LimiterCoeff()*config->GetRefElemLength();
225  if ( geometry->node[iPoint]->GetSharpEdge_Distance() < config->GetAdjSharp_LimiterCoeff()*eps )
226  Sensitivity = 0.0;
227  }
228 
229  /*--- Store the sensitivities ---*/
230  if (!time_stepping) {
231  solver[ADJFLOW_SOL]->GetNodes()->SetSensitivity(iPoint, iDim, Sensitivity);
232  } else {
233  solver[ADJFLOW_SOL]->GetNodes()->SetSensitivity(iPoint, iDim,
234  solver[ADJFLOW_SOL]->GetNodes()->GetSensitivity(iPoint, iDim) + Sensitivity);
235  }
236  }
237  }
238  solver[ADJFLOW_SOL]->SetSurface_Sensitivity(geometry, config);
