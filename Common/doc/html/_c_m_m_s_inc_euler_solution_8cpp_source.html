95 void CMMSIncEulerSolution::GetSolution(const su2double *val_coords,
96  const su2double val_t,
97  su2double *val_solution) {
98 
99  /* Easier storage of the x- and y-coordinates. */
100  const su2double x = val_coords[0];
101  const su2double y = val_coords[1];
102 
103  /* Compute the primitives from the defined solution. */
104  const su2double u = u_0*(sin(x*x + y*y) + epsilon);
105  const su2double v = v_0*(cos(x*x + y*y) + epsilon);
106  const su2double p = P_0*(sin(x*x + y*y) + 2.0);
107 
108  /* For the incompressible solver, we return the primitive variables
109  directly, as they are used for the working variables in the solver.
110  Note that the implementation below is valid for both 2D and 3D. */
111  val_solution[0] = p;
112  val_solution[1] = u;
113  val_solution[2] = v;
114  val_solution[3] = 0.0;
115  val_solution[nVar-1] = Temperature;
116 
117 }

119 void CMMSIncEulerSolution::GetMMSSourceTerm(const su2double *val_coords,
120  const su2double val_t,
121  su2double *val_source) {
122 
123  /*--- Easier storage of the x- and y-coordinates. ---*/
124  const su2double x = val_coords[0];
125  const su2double y = val_coords[1];
126 
127  /*--- The expressions for the source terms are generated
128  automatically by the sympy package in python.---*/
129  val_source[0] = 2*Density*(u_0*x*cos(pow(x, 2) + pow(y, 2)) - v_0*y*sin(pow(x, 2) + pow(y, 2)));
130  val_source[1] = 4*Density*pow(u_0, 2)*x*(epsilon + sin(pow(x, 2) + pow(y, 2)))*cos(pow(x, 2) + pow(y, 2)) - 2*Density*u_0*v_0*y*(epsilon + sin(pow(x, 2) + pow(y, 2)))*sin(pow(x, 2) + pow(y, 2)) + 2*Density*u_0*v_0*y*(epsilon + cos(pow(x, 2) + pow(y, 2)))*cos(pow(x, 2) + pow(y, 2)) + 2*P_0*x*cos(pow(x, 2) + pow(y, 2));
131  val_source[2] = -2*Density*u_0*v_0*x*(epsilon + sin(pow(x, 2) + pow(y, 2)))*sin(pow(x, 2) + pow(y, 2)) + 2*Density*u_0*v_0*x*(epsilon + cos(pow(x, 2) + pow(y, 2)))*cos(pow(x, 2) + pow(y, 2)) - 4*Density*pow(v_0, 2)*y*(epsilon + cos(pow(x, 2) + pow(y, 2)))*sin(pow(x, 2) + pow(y, 2)) + 2*P_0*y*cos(pow(x, 2) + pow(y, 2));
132  val_source[3] = 0.0;
133  val_source[nVar-1] = 0.0;
134 
135 }
