199 void CFEALinearElasticity::Compute_Constitutive_Matrix(CElement *element_container, CConfig *config) {
200 
201  /*--- Compute the D Matrix (for plane stress and 2-D)---*/
202 
203 
204  if (nDim == 2) {
205  if (plane_stress) {
206  /*--- We enable plane stress cases ---*/
207 
208  D_Mat[0][0] = E/(1-Nu*Nu); D_Mat[0][1] = (E*Nu)/(1-Nu*Nu); D_Mat[0][2] = 0.0;
209  D_Mat[1][0] = (E*Nu)/(1-Nu*Nu); D_Mat[1][1] = E/(1-Nu*Nu); D_Mat[1][2] = 0.0;
210  D_Mat[2][0] = 0.0; D_Mat[2][1] = 0.0; D_Mat[2][2] = ((1-Nu)*E)/(2*(1-Nu*Nu));
211  }
212  else {
213  /*--- Assuming plane strain as a general case ---*/
214 
215  D_Mat[0][0] = Lambda + 2.0*Mu; D_Mat[0][1] = Lambda; D_Mat[0][2] = 0.0;
216  D_Mat[1][0] = Lambda; D_Mat[1][1] = Lambda + 2.0*Mu; D_Mat[1][2] = 0.0;
217  D_Mat[2][0] = 0.0; D_Mat[2][1] = 0.0; D_Mat[2][2] = Mu;
218  }
219 
220  }
221  else {
222 
223  su2double Lbda_2Mu = Lambda + 2.0*Mu;
224 
225  D_Mat[0][0] = Lbda_2Mu; D_Mat[0][1] = Lambda; D_Mat[0][2] = Lambda; D_Mat[0][3] = 0.0; D_Mat[0][4] = 0.0; D_Mat[0][5] = 0.0;
226  D_Mat[1][0] = Lambda; D_Mat[1][1] = Lbda_2Mu; D_Mat[1][2] = Lambda; D_Mat[1][3] = 0.0; D_Mat[1][4] = 0.0; D_Mat[1][5] = 0.0;
227  D_Mat[2][0] = Lambda; D_Mat[2][1] = Lambda; D_Mat[2][2] = Lbda_2Mu; D_Mat[2][3] = 0.0; D_Mat[2][4] = 0.0; D_Mat[2][5] = 0.0;
228  D_Mat[3][0] = 0.0; D_Mat[3][1] = 0.0; D_Mat[3][2] = 0.0; D_Mat[3][3] = Mu; D_Mat[3][4] = 0.0; D_Mat[3][5] = 0.0;

319  }
320  }
321

329  }

332  * while in the output it is the 4th component for practical reasons. ---*/
333  element->Add_NodalStress(iNode, 0, Stress[0] * Ni_Extrap);
334  element->Add_NodalStress(iNode, 1, Stress[1] * Ni_Extrap);
335  element->Add_NodalStress(iNode, 2, Stress[3] * Ni_Extrap);
336  element->Add_NodalStress(iNode, 3, Stress[2] * Ni_Extrap);
337  element->Add_NodalStress(iNode, 4, Stress[4] * Ni_Extrap);
338  element->Add_NodalStress(iNode, 5, Stress[5] * Ni_Extrap);
339  }
340  }
