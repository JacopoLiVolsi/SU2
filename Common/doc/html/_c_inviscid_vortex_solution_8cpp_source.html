107 void CInviscidVortexSolution::GetSolution(const su2double *val_coords,
108  const su2double val_t,
109  su2double *val_solution) {
110 
111  /* Compute the free stream velocities in x- and y-direction. */
112  const su2double VelInf = MachVortex*sqrt(Gamma);
113  const su2double uInf = VelInf*cos(thetaVortex*PI_NUMBER/180.0);
114  const su2double vInf = VelInf*sin(thetaVortex*PI_NUMBER/180.0);
115 
116  /* Compute the coordinates relative to the center of the vortex. */
117  const su2double dx = val_coords[0] - (x0Vortex + val_t*uInf);
118  const su2double dy = val_coords[1] - (y0Vortex + val_t*vInf);
119 
120  /* Compute the components of the velocity. */
121  su2double f = 1.0 - (dx*dx + dy*dy)/(RVortex*RVortex);
122  su2double t1 = epsVortex*dy*exp(0.5*f)/(2.0*PI_NUMBER*RVortex);
123  su2double u = uInf - VelInf*t1;
124 
125  t1 = epsVortex*dx*exp(0.5*f)/(2.0*PI_NUMBER*RVortex);
126  su2double v = vInf + VelInf*t1;
127 
128  /* Compute the density and the pressure. */
129  t1 = 1.0 - epsVortex*epsVortex*Gm1
130  * MachVortex*MachVortex*exp(f)/(8.0*PI_NUMBER*PI_NUMBER);
131 
132  su2double rho = pow(t1,ovGm1);
133  su2double p = pow(t1,gamOvGm1);
134 
135  /* Compute the conservative variables. Note that both 2D and 3D
136  cases are treated correctly. */
137  val_solution[0] = rho;
138  val_solution[1] = rho*u;
139  val_solution[2] = rho*v;
140  val_solution[3] = 0.0;
141  val_solution[nVar-1] = p*ovGm1 + 0.5*rho*(u*u + v*v);
142 }
