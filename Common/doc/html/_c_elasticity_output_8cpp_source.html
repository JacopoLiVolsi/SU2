102 void CElasticityOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
103 
104  CSolver* fea_solver = solver[FEA_SOL];
105 
106  /*--- Residuals: ---*/
107  /*--- Linear analysis: RMS of the displacements in the nDim coordinates ---*/
108  /*--- Nonlinear analysis: UTOL, RTOL and DTOL (defined in the Postprocessing function) ---*/
109 
110 
111  if (linear_analysis){
112  SetHistoryOutputValue("RMS_DISP_X", log10(fea_solver->GetRes_RMS(0)));
113  SetHistoryOutputValue("RMS_DISP_Y", log10(fea_solver->GetRes_RMS(1)));
114  if (nDim == 3){
115  SetHistoryOutputValue("RMS_DISP_Z", log10(fea_solver->GetRes_RMS(2)));
116  }
117  } else if (nonlinear_analysis){
118  SetHistoryOutputValue("RMS_UTOL", log10(fea_solver->LinSysSol.norm()));
119  SetHistoryOutputValue("RMS_RTOL", log10(fea_solver->LinSysRes.norm()));
120  SetHistoryOutputValue("RMS_ETOL", log10(dotProd(fea_solver->LinSysSol, fea_solver->LinSysRes)));
121 
122  }
123 
124  if (multiZone){
125  SetHistoryOutputValue("BGS_DISP_X", log10(fea_solver->GetRes_BGS(0)));
126  SetHistoryOutputValue("BGS_DISP_Y", log10(fea_solver->GetRes_BGS(1)));
127  if (nDim == 3) SetHistoryOutputValue("BGS_DISP_Z", log10(fea_solver->GetRes_BGS(2)));
128  }
129 
130  SetHistoryOutputValue("VMS", fea_solver->GetTotal_CFEA());
131  SetHistoryOutputValue("LOAD_INCREMENT", fea_solver->GetLoad_Increment());
132  SetHistoryOutputValue("LOAD_RAMP", fea_solver->GetForceCoeff());
133 
134  SetHistoryOutputValue("LINSOL_ITER", fea_solver->GetIterLinSolver());
135  SetHistoryOutputValue("LINSOL_RESIDUAL", log10(fea_solver->GetResLinSolver()));
