815 template<class ScalarType>
816 void CSysMatrix<ScalarType>::BuildILUPreconditioner(bool transposed) {
817 
818  unsigned long index, index_, iVar;
819  ScalarType *Block_ij;
820  const ScalarType *Block_jk;
821  long iPoint, jPoint, kPoint;
822 
823  /*--- Copy block matrix, note that the original matrix
824  is modified by the algorithm, so that we have the factorization stored
825  in the ILUMatrix at the end of this preprocessing. ---*/
826 
827  for (iVar = 0; iVar < nnz_ilu*nVar*nEqn; iVar++) ILU_matrix[iVar] = 0.0;
828 
829  for (iPoint = 0; iPoint < (long)nPointDomain; iPoint++) {
830  for (index = row_ptr[iPoint]; index < row_ptr[iPoint+1]; index++) {
831  jPoint = col_ind[index];
832  if (transposed) {
833  Block_ij = GetBlock(jPoint, iPoint);
834  SetBlockTransposed_ILUMatrix(iPoint, jPoint, Block_ij);
835  } else {
836  Block_ij = GetBlock(iPoint, jPoint);
837  SetBlock_ILUMatrix(iPoint, jPoint, Block_ij);
838  }
839  }
840  }
841 
842  /*--- Transform system in Upper Matrix ---*/
843 
844  for (iPoint = 1; iPoint < (long)nPointDomain; iPoint++) {
845 
846  /*--- Invert and store the previous diagonal block to later compute the weight. ---*/
847 
848  InverseDiagonalBlock_ILUMatrix(iPoint-1, &invM[(iPoint-1)*nVar*nVar]);
849 
850  /*--- For each row (unknown), loop over all entries in A on this row
851  row_ptr_ilu[iPoint+1] will have the index for the first entry on the next
852  row. ---*/
853 
854  for (index = row_ptr_ilu[iPoint]; index < row_ptr_ilu[iPoint+1]; index++) {
855 
856  /*--- jPoint here is the column for each entry on this row ---*/
857 
858  jPoint = col_ind_ilu[index];
859 
860  /*--- Check that this column is in the lower triangular portion ---*/
861 
862  if (jPoint < iPoint) {
863 
864  /*--- If we're in the lower triangle, multiply the block by
865  the inverse of the corresponding diagonal block. ---*/
866 
867  Block_ij = &ILU_matrix[index*nVar*nEqn];
868  MatrixMatrixProduct(Block_ij, &invM[jPoint*nVar*nVar], block_weight);
869 
870  /*--- block_weight holds Aij*inv(Ajj). Jump to the row for jPoint ---*/
871 
872  for (index_ = row_ptr_ilu[jPoint]; index_ < row_ptr_ilu[jPoint+1]; index_++) {
873 
874  /*--- Get the column of the entry ---*/
875 
876  kPoint = col_ind_ilu[index_];
877 
878  /*--- If the column is greater than or equal to jPoint, i.e., the
879  upper triangular part, then multiply and modify the matrix.
880  Here, Aik' = Aik - Aij*inv(Ajj)*Ajk. ---*/
881 
882  if (kPoint > jPoint) {
883 
884  Block_jk = &ILU_matrix[index_*nVar*nEqn];
885  MatrixMatrixProduct(block_weight, Block_jk, block);
886  SubtractBlock_ILUMatrix(iPoint, kPoint, block);
887 
888  }
889  }
890 
891  /*--- Lastly, store block_weight in the lower triangular part, which
892  will be reused during the forward solve in the precon/smoother. ---*/
893 
894  for (iVar = 0; iVar < nVar*nEqn; ++iVar)
895  Block_ij[iVar] = block_weight[iVar];
896 
897  }
898  }
899  }
900 
901  InverseDiagonalBlock_ILUMatrix(nPointDomain-1, &invM[(nPointDomain-1)*nVar*nVar]);
902 
903 }

906 void CSysMatrix<ScalarType>::ComputeILUPreconditioner(const CSysVector<ScalarType> & vec, CSysVector<ScalarType> & prod, CGeometry *geometry, CConfig *config) {
907 
908  unsigned long index, iVar;
909  const ScalarType *Block_ij;
910  long iPoint, jPoint;
911 
912  /*--- Copy vector to then work on prod in place ---*/
913 
914  for (iPoint = 0; iPoint < long(nPointDomain*nVar); iPoint++)
915  prod[iPoint] = vec[iPoint];
916 
917  /*--- Forward solve the system using the lower matrix entries that
918  were computed and stored during the ILU preprocessing. Note
919  that we are overwriting the residual vector as we go. ---*/
920 
921  for (iPoint = 1; iPoint < (long)nPointDomain; iPoint++) {
922  for (index = row_ptr_ilu[iPoint]; index < row_ptr_ilu[iPoint+1]; index++) {

927  }
928  }
929  }
930

932

940  if ((jPoint >= iPoint+1) && (jPoint < (long)nPointDomain)) {
941  Block_ij = &ILU_matrix[index*nVar*nEqn];
942  MatrixVectorProductSub(Block_ij, &prod[jPoint*nVar], sum_vector);
943  }
944  }
945 
946  MatrixVectorProduct(&invM[iPoint*nVar*nVar], sum_vector, &prod[iPoint*nVar]);
947  }
948 
949  /*--- MPI Parallelization ---*/
950 
951  InitiateComms(prod, geometry, config, SOLUTION_MATRIX);
952  CompleteComms(prod, geometry, config, SOLUTION_MATRIX);

956 template<class ScalarType>
957 void CSysMatrix<ScalarType>::ComputeLU_SGSPreconditioner(const CSysVector<ScalarType> & vec, CSysVector<ScalarType> & prod, CGeometry *geometry, CConfig *config) {
958  unsigned long iPoint, iVar;
959 
960  /*--- First part of the symmetric iteration: (D+L).x* = b ---*/
961 
962  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
963  LowerProduct(prod, iPoint); // Compute L.x*
964  for (iVar = 0; iVar < nVar; iVar++)
965  prod[iPoint*nVar+iVar] = vec[iPoint*nVar+iVar] - prod_row_vector[iVar]; // Compute aux_vector = b - L.x*
966  Gauss_Elimination(iPoint, &prod[iPoint*nVar]); // Solve D.x* = aux_vector
967  }
968 
969  /*--- MPI Parallelization ---*/
970 
971  InitiateComms(prod, geometry, config, SOLUTION_MATRIX);
972  CompleteComms(prod, geometry, config, SOLUTION_MATRIX);
973 
974  /*--- Second part of the symmetric iteration: (D+U).x_(1) = D.x* ---*/
975 
976  for (iPoint = nPointDomain-1; (int)iPoint >= 0; iPoint--) {
977  DiagonalProduct(prod, iPoint); // Compute D.x*
978  for (iVar = 0; iVar < nVar; iVar++)
979  aux_vector[iVar] = prod_row_vector[iVar]; // Compute aux_vector = D.x*
980  UpperProduct(prod, iPoint); // Compute U.x_(n+1)
981  for (iVar = 0; iVar < nVar; iVar++)
982  prod[iPoint*nVar+iVar] = aux_vector[iVar] - prod_row_vector[iVar]; // Compute aux_vector = D.x*-U.x_(n+1)
983  Gauss_Elimination(iPoint, &prod[iPoint*nVar]); // Solve D.x* = aux_vector
984  }
985 
986  /*--- MPI Parallelization ---*/
987 
988  InitiateComms(prod, geometry, config, SOLUTION_MATRIX);
989  CompleteComms(prod, geometry, config, SOLUTION_MATRIX);
990
