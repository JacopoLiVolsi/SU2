164 void CFlowCompFEMOutput::LoadVolumeDataFEM(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iElem, unsigned long index, unsigned short dof){
165 
166  unsigned short iDim;
167 
168  unsigned short nVar = solver[FLOW_SOL]->GetnVar();
169 
170  /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
171  geometrical information for the FEM DG solver. ---*/
172 
173  CMeshFEM_DG *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
174 
175  CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
176 
177  /*--- Get a pointer to the fluid model class from the DG-FEM solver
178  so that we can access the states below. ---*/
179 
180  CFluidModel *DGFluidModel = solver[FLOW_SOL]->GetFluidModel();
181 
182  /* Set the pointers for the solution for this element. */
183 
184  const unsigned long offset = nVar*volElem[iElem].offsetDOFsSolLocal;
185  su2double *solDOFs = solver[FLOW_SOL]->GetVecSolDOFs() + offset;
186 
187  /*--- Get the conservative variables for this particular DOF. ---*/
188 
189  const su2double *U = solDOFs+dof*nVar;
190 
191  /*--- Load the coordinate values of the solution DOFs. ---*/
192 
193  const su2double *coor = volElem[iElem].coorSolDOFs.data() + dof*nDim;
194 
195  /*--- Prepare the primitive states. ---*/
196 
197  const su2double DensityInv = 1.0/U[0];
198  su2double vel[3], Velocity2 = 0.0;
199  for(iDim=0; iDim<nDim; ++iDim) {
200  vel[iDim] = U[iDim+1]*DensityInv;
201  Velocity2 += vel[iDim]*vel[iDim];
202  }
203  su2double StaticEnergy = U[nDim+1]*DensityInv - 0.5*Velocity2;
204  DGFluidModel->SetTDState_rhoe(U[0], StaticEnergy);
205 
206 
207  SetVolumeOutputValue("COORD-X", index, coor[0]);
208  SetVolumeOutputValue("COORD-Y", index, coor[1]);
209  if (nDim == 3)
210  SetVolumeOutputValue("COORD-Z", index, coor[2]);
211  SetVolumeOutputValue("DENSITY", index, U[0]);
212  SetVolumeOutputValue("MOMENTUM-X", index, U[1]);
213  SetVolumeOutputValue("MOMENTUM-Y", index, U[2]);
214  if (nDim == 3){
215  SetVolumeOutputValue("MOMENTUM-Z", index, U[3]);
216  SetVolumeOutputValue("ENERGY", index, U[4]);
217  } else {
218  SetVolumeOutputValue("ENERGY", index, U[3]);
219  }
220 
221  SetVolumeOutputValue("PRESSURE", index, DGFluidModel->GetPressure());
222  SetVolumeOutputValue("TEMPERATURE", index, DGFluidModel->GetTemperature());
223  SetVolumeOutputValue("MACH", index, sqrt(Velocity2)/DGFluidModel->GetSoundSpeed());
224  SetVolumeOutputValue("PRESSURE_COEFF", index, DGFluidModel->GetCp());
225 
226  if (config->GetKind_Solver() == FEM_NAVIER_STOKES){
227  SetVolumeOutputValue("LAMINAR_VISCOSITY", index, DGFluidModel->GetLaminarViscosity());
228  }
229  if ((config->GetKind_Solver() == FEM_LES) && (config->GetKind_SGS_Model() != IMPLICIT_LES)){
230  // todo: Export Eddy instead of Laminar viscosity
231  SetVolumeOutputValue("EDDY_VISCOSITY", index, DGFluidModel->GetLaminarViscosity());
232  }
