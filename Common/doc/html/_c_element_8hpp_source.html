475  static_assert(NDIM==2 || NDIM==3, "ComputeGrad_impl expects 2D or 3D");
476

478  su2double dNiXj[NGAUSS][NNODE][NDIM];

491  template<FrameType FRAME>
492  void ComputeGrad_impl(void) {
493 
494  su2double Jacobian[NDIM][NDIM], ad[NDIM][NDIM];
495  su2double detJac, GradNi_Xj;
496  unsigned short iNode, iDim, jDim, iGauss;
497 
498  /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
499  for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed) ---*/
500  const su2activematrix& Coord = (FRAME==REFERENCE) ? RefCoord : CurrentCoord;
501 
502  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
503 
504  /*--- Jacobian transformation ---*/
505  /*--- This does dX/dXi transpose ---*/
506 
507  for (iDim = 0; iDim < NDIM; iDim++)
508  for (jDim = 0; jDim < NDIM; jDim++)
509  Jacobian[iDim][jDim] = 0.0;
510 
511  for (iNode = 0; iNode < NNODE; iNode++)
512  for (iDim = 0; iDim < NDIM; iDim++)
513  for (jDim = 0; jDim < NDIM; jDim++)
514  Jacobian[iDim][jDim] += Coord(iNode,jDim) * dNiXj[iGauss][iNode][iDim];
515 
516  if (NDIM == 2) {
517  /*--- Adjoint to Jacobian ---*/
518 
519  ad[0][0] = Jacobian[1][1]; ad[0][1] = -Jacobian[0][1];
520  ad[1][0] = -Jacobian[1][0]; ad[1][1] = Jacobian[0][0];
521 
522  /*--- Determinant of Jacobian ---*/
523 
524  detJac = ad[0][0]*ad[1][1]-ad[0][1]*ad[1][0];
525  }
526  else {
527  /*--- Adjoint to Jacobian ---*/
528 
529  ad[0][0] = Jacobian[1][1]*Jacobian[2][2]-Jacobian[1][2]*Jacobian[2][1];
530  ad[0][1] = Jacobian[0][2]*Jacobian[2][1]-Jacobian[0][1]*Jacobian[2][2];
531  ad[0][2] = Jacobian[0][1]*Jacobian[1][2]-Jacobian[0][2]*Jacobian[1][1];
532  ad[1][0] = Jacobian[1][2]*Jacobian[2][0]-Jacobian[1][0]*Jacobian[2][2];
533  ad[1][1] = Jacobian[0][0]*Jacobian[2][2]-Jacobian[0][2]*Jacobian[2][0];
534  ad[1][2] = Jacobian[0][2]*Jacobian[1][0]-Jacobian[0][0]*Jacobian[1][2];
535  ad[2][0] = Jacobian[1][0]*Jacobian[2][1]-Jacobian[1][1]*Jacobian[2][0];
536  ad[2][1] = Jacobian[0][1]*Jacobian[2][0]-Jacobian[0][0]*Jacobian[2][1];
537  ad[2][2] = Jacobian[0][0]*Jacobian[1][1]-Jacobian[0][1]*Jacobian[1][0];
538 
539  /*--- Determinant of Jacobian ---*/
540 
541  detJac = Jacobian[0][0]*ad[0][0]+Jacobian[0][1]*ad[1][0]+Jacobian[0][2]*ad[2][0];
542  }
543 
544  if (FRAME==REFERENCE)
545  GaussPoint[iGauss].SetJ_X(detJac);
546  else
547  GaussPoint[iGauss].SetJ_x(detJac);
548 
549  /*--- Jacobian inverse (it was already computed as transpose) ---*/
550 
551  for (iDim = 0; iDim < NDIM; iDim++)
552  for (jDim = 0; jDim < NDIM; jDim++)
553  Jacobian[iDim][jDim] = ad[iDim][jDim]/detJac;
554 
555  /*--- Derivatives with respect to global coordinates ---*/
556 
557  for (iNode = 0; iNode < NNODE; iNode++) {
558  for (iDim = 0; iDim < NDIM; iDim++) {
559  GradNi_Xj = 0.0;
560  for (jDim = 0; jDim < NDIM; jDim++)
561  GradNi_Xj += Jacobian[iDim][jDim] * dNiXj[iGauss][iNode][jDim];
562 
563  if (FRAME==REFERENCE)
564  GaussPoint[iGauss].SetGradNi_Xj(GradNi_Xj, iDim, iNode);
565  else
566  GaussPoint[iGauss].SetGradNi_xj(GradNi_Xj, iDim, iNode);
567  }
568  }
569 
570  }
571 
572  }
