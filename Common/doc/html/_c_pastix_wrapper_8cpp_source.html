172  else
173  aux[j-begin].first = pastix_int_t(map[col_ind[j]-nPointDomain]+1);
174  aux[j-begin].second = j;
175  }
176  sort(aux.begin(), aux.end());
177 
178  for (j = 0; j < nnz_row; ++j)
179  {
180  rowidx.push_back(aux[j].first);
181  sort_order.back()[j] = aux[j].second;
182  }
183  }
184  else
185  {
186  /*--- These are all internal, no need to go through map. ---*/
187  for (j = begin; j < end; ++j)

189  }
