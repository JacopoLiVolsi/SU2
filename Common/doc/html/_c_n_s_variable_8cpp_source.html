47 bool CNSVariable::SetVorticity_StrainMag() {
48 
49  for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
50 
51  /*--- Vorticity ---*/
52 
53  Vorticity(iPoint,0) = 0.0; Vorticity(iPoint,1) = 0.0;
54 
55  Vorticity(iPoint,2) = Gradient_Primitive(iPoint,2,0)-Gradient_Primitive(iPoint,1,1);
56 
57  if (nDim == 3) {
58  Vorticity(iPoint,0) = Gradient_Primitive(iPoint,3,1)-Gradient_Primitive(iPoint,2,2);
59  Vorticity(iPoint,1) = -(Gradient_Primitive(iPoint,3,0)-Gradient_Primitive(iPoint,1,2));
60  }
61 
62  /*--- Strain Magnitude ---*/
63 
64  AD::StartPreacc();
65  AD::SetPreaccIn(Gradient_Primitive[iPoint], nDim+1, nDim);
66 
67  su2double Div = 0.0;
68  for (unsigned long iDim = 0; iDim < nDim; iDim++)
69  Div += Gradient_Primitive(iPoint,iDim+1,iDim);
70 
71  StrainMag(iPoint) = 0.0;
72 
73  /*--- Add diagonal part ---*/
74 
75  for (unsigned long iDim = 0; iDim < nDim; iDim++) {
76  StrainMag(iPoint) += pow(Gradient_Primitive(iPoint,iDim+1,iDim) - 1.0/3.0*Div, 2.0);
77  }
78  if (nDim == 2) {
79  StrainMag(iPoint) += pow(1.0/3.0*Div, 2.0);
80  }
81 
82  /*--- Add off diagonals ---*/
83 
84  StrainMag(iPoint) += 2.0*pow(0.5*(Gradient_Primitive(iPoint,1,1) + Gradient_Primitive(iPoint,2,0)), 2);
85 
86  if (nDim == 3) {
87  StrainMag(iPoint) += 2.0*pow(0.5*(Gradient_Primitive(iPoint,1,2) + Gradient_Primitive(iPoint,3,0)), 2);
88  StrainMag(iPoint) += 2.0*pow(0.5*(Gradient_Primitive(iPoint,2,2) + Gradient_Primitive(iPoint,3,1)), 2);
89  }
90 
91  StrainMag(iPoint) = sqrt(2.0*StrainMag(iPoint));
92 
93  AD::SetPreaccOut(StrainMag(iPoint));
94  AD::EndPreacc();
95  }
96  return false;
97 }

99 void CNSVariable::SetRoe_Dissipation_NTS(unsigned long iPoint,
100  su2double val_delta,
101  su2double val_const_DES){
102 
103  static const su2double cnu = pow(0.09, 1.5),
104  ch1 = 3.0,
105  ch2 = 1.0,
106  ch3 = 2.0,
107  sigma_max = 1.0;
108 
109  unsigned long iDim;
110  su2double Omega, Omega_2 = 0.0, Baux, Gaux, Lturb, Kaux, Aaux;
111 
112  AD::StartPreacc();
113  AD::SetPreaccIn(Vorticity[iPoint], 3);
114  AD::SetPreaccIn(StrainMag(iPoint));
115  AD::SetPreaccIn(val_delta);
116  AD::SetPreaccIn(val_const_DES);
117  /*--- Density ---*/
118  AD::SetPreaccIn(Solution(iPoint,0));
119  /*--- Laminar viscosity --- */
120  AD::SetPreaccIn(Primitive(iPoint,nDim+5));
121  /*--- Eddy viscosity ---*/
122  AD::SetPreaccIn(Primitive(iPoint,nDim+6));
123 
124  /*--- Central/upwind blending based on:
125  * Zhixiang Xiao, Jian Liu, Jingbo Huang, and Song Fu. "Numerical
126  * Dissipation Effects on Massive Separation Around Tandem Cylinders",
127  * AIAA Journal, Vol. 50, No. 5 (2012), pp. 1119-1136.
128  * https://doi.org/10.2514/1.J051299
129  * ---*/
130 
131  for (iDim = 0; iDim < 3; iDim++){
132  Omega_2 += pow(Vorticity(iPoint,iDim),2);
133  }
134  Omega = sqrt(Omega_2);
135 
136  Baux = (ch3 * Omega * max(StrainMag(iPoint), Omega)) /
137  max((pow(StrainMag(iPoint),2)+Omega_2)*0.5, 1E-20);
138  Gaux = tanh(pow(Baux,4.0));
139 
140  Kaux = max(sqrt((Omega_2 + pow(StrainMag(iPoint), 2))*0.5), 0.1 * inv_TimeScale);
141 
142  const su2double nu = GetLaminarViscosity(iPoint)/GetDensity(iPoint);
143  const su2double nu_t = GetEddyViscosity(iPoint)/GetDensity(iPoint);
144  Lturb = sqrt((nu + nu_t)/(cnu*Kaux));
145 
146  Aaux = ch2*max((val_const_DES*val_delta/Lturb)/Gaux - 0.5, 0.0);
147 
148  Roe_Dissipation(iPoint) = sigma_max * tanh(pow(Aaux, ch1));
149 
150  AD::SetPreaccOut(Roe_Dissipation(iPoint));
151  AD::EndPreacc();
152 
153 }

155 void CNSVariable::SetRoe_Dissipation_FD(unsigned long iPoint, su2double val_wall_dist){
156 
157  /*--- Constants for Roe Dissipation ---*/
158 
159  const passivedouble k2 = pow(0.41,2.0);
160 
161  AD::StartPreacc();
162  AD::SetPreaccIn(Gradient_Primitive[iPoint], nVar, nDim);
163  AD::SetPreaccIn(val_wall_dist);
164  /*--- Eddy viscosity ---*/
165  AD::SetPreaccIn(Primitive(iPoint,nDim+5));
166  /*--- Laminar viscosity --- */
167  AD::SetPreaccIn(Primitive(iPoint,nDim+6));
168 
169  su2double uijuij = 0.0;
170 
171  for(unsigned long iDim = 0; iDim < nDim; ++iDim)
172  for(unsigned long jDim = 0; jDim < nDim; ++jDim)
173  uijuij += pow(Gradient_Primitive(iPoint,1+iDim,jDim),2);
174 
175  uijuij = max(sqrt(uijuij),1e-10);
176 
177  const su2double nu = GetLaminarViscosity(iPoint)/GetDensity(iPoint);
178  const su2double nu_t = GetEddyViscosity(iPoint)/GetDensity(iPoint);
179  const su2double r_d = (nu + nu_t)/(uijuij*k2*pow(val_wall_dist,2));
180 
181  Roe_Dissipation(iPoint) = 1.0-tanh(pow(8.0*r_d,3.0));
182 
183  AD::SetPreaccOut(Roe_Dissipation(iPoint));
184  AD::EndPreacc();
185 }
