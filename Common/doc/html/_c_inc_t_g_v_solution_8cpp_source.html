108 void CIncTGVSolution::GetSolution(const su2double *val_coords,
109  const su2double val_t,
110  su2double *val_solution) {
111 
112  /* The exact solution is set for the incompressible Taylor-Green
113  vortex case. This is the classic solution from the original work
114  of Taylor and Green for the specific 2D situation where the
115  exact solution can be derived for an incompressible flow. */
116 
117  /* Store the termporal term more easily (Taylor expansion). */
118  su2double F = 1.0 - 2.0*(tgvViscosity/tgvDensity)*val_t;
119 
120  /* Compute the primitive variables. */
121  su2double u = tgvVelocity * F * (sin(val_coords[0]/tgvLength)*
122  cos(val_coords[1]/tgvLength));
123  su2double v = -tgvVelocity * F * (cos(val_coords[0]/tgvLength)*
124  sin(val_coords[1]/tgvLength));
125 
126  su2double B = (cos(2.0*val_coords[0]/tgvLength) +
127  cos(2.0*val_coords[1]/tgvLength));
128  su2double p = -(tgvDensity/4.0)*B*F*F;
129 
130  /* Compute the conservative variables. Note that both 2D and 3D
131  cases are treated correctly. */
132  val_solution[0] = p;
133  val_solution[1] = u;
134  val_solution[2] = v;
135  val_solution[nVar-1] = Temperature;
136 }
