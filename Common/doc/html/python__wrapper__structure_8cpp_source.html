341 bool CDriver::ComputeVertexForces(unsigned short iMarker, unsigned long iVertex) {
342 
343  unsigned long iPoint;
344  unsigned short iDim, jDim;
345  su2double *Normal, AreaSquare, Area;
346  bool halo;
347 
348  unsigned short FinestMesh = config_container[ZONE_0]->GetFinestMesh();
349 
350  /*--- Check the kind of fluid problem ---*/
351  bool compressible = (config_container[ZONE_0]->GetKind_Regime() == COMPRESSIBLE);
352  bool incompressible = (config_container[ZONE_0]->GetKind_Regime() == INCOMPRESSIBLE);
353  bool viscous_flow = ((config_container[ZONE_0]->GetKind_Solver() == NAVIER_STOKES) ||
354  (config_container[ZONE_0]->GetKind_Solver() == RANS) );
355 
356  /*--- Parameters for the calculations ---*/
357  // Pn: Pressure
358  // Pinf: Pressure_infinite
359  // div_vel: Velocity divergence
360  // Dij: Dirac delta
361  su2double Pn = 0.0, div_vel = 0.0, Dij = 0.0;
362  su2double Viscosity = 0.0;
363  su2double Grad_Vel[3][3] = { {0.0, 0.0, 0.0} ,
364  {0.0, 0.0, 0.0} ,
365  {0.0, 0.0, 0.0} } ;
366  su2double Tau[3][3] = { {0.0, 0.0, 0.0} ,
367  {0.0, 0.0, 0.0} ,
368  {0.0, 0.0, 0.0} } ;
369 
370  su2double Pinf = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetPressure_Inf();
371 
372  iPoint = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNode();
373 
374  /*--- It is necessary to distinguish the halo nodes from the others, since they introduce non physical forces. ---*/
375  if(geometry_container[ZONE_0][INST_0][MESH_0]->node[iPoint]->GetDomain()) {
376  /*--- Get the normal at the vertex: this normal goes inside the fluid domain. ---*/
377  Normal = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNormal();
378  AreaSquare = 0.0;
379  for(iDim = 0; iDim < nDim; iDim++) {
380  AreaSquare += Normal[iDim]*Normal[iDim];
381  }
382  Area = sqrt(AreaSquare);
383 
384  /*--- Get the values of pressure and viscosity ---*/
385  Pn = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetNodes()->GetPressure(iPoint);
386  if (viscous_flow) {
387  for(iDim=0; iDim<nDim; iDim++) {
388  for(jDim=0; jDim<nDim; jDim++) {
389  Grad_Vel[iDim][jDim] = solver_container[ZONE_0][INST_0][FinestMesh][FLOW_SOL]->GetNodes()->GetGradient_Primitive(iPoint, iDim+1, jDim);
390  }
391  }
392  Viscosity = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
393  }
394 
395  /*--- Calculate the inviscid (pressure) part of tn in the fluid nodes (force units) ---*/
396  for (iDim = 0; iDim < nDim; iDim++) {
397  PyWrapNodalForce[iDim] = -(Pn-Pinf)*Normal[iDim]; //NB : norm(Normal) = Area
398  }
399 
400  /*--- Calculate the viscous (shear stress) part of tn in the fluid nodes (force units) ---*/
401  if ((incompressible || compressible) && viscous_flow) {
402  div_vel = 0.0;
403  for (iDim = 0; iDim < nDim; iDim++)
404  div_vel += Grad_Vel[iDim][iDim];
405  if (incompressible) div_vel = 0.0;
406 
407  for (iDim = 0; iDim < nDim; iDim++) {
408  for (jDim = 0 ; jDim < nDim; jDim++) {
409  Dij = 0.0; if (iDim == jDim) Dij = 1.0;
410  Tau[iDim][jDim] = Viscosity*(Grad_Vel[jDim][iDim] + Grad_Vel[iDim][jDim]) - TWO3*Viscosity*div_vel*Dij;
411  PyWrapNodalForce[iDim] += Tau[iDim][jDim]*Normal[jDim];
412  }
413  }
414  }
415 
416  //Divide by local are in case of force density communication.
417  for(iDim = 0; iDim < nDim; iDim++) {
418  PyWrapNodalForceDensity[iDim] = PyWrapNodalForce[iDim]/Area;
419  }
420 
421  halo = false;
422  }
423  else {
424  halo = true;
425  }
426 
427  return halo;
428

591  su2double Gamma = config_container[ZONE_0]->GetGamma();
592  su2double Gamma_Minus_One = Gamma - 1.0;
593  su2double Cp = (Gamma / Gamma_Minus_One) * Gas_Constant;
594  su2double Area;
595  su2double laminar_viscosity, thermal_conductivity, dTdn;
596  su2double *Normal, GradT[3] = {0.0,0.0,0.0}, UnitNormal[3] = {0.0,0.0,0.0};
597 
598  bool compressible = (config_container[ZONE_0]->GetKind_Regime() == COMPRESSIBLE);
599 
600  vertexWallHeatFlux = 0.0;
601  dTdn = 0.0;
602 
603  iPoint = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNode();
604 
605  if(geometry_container[ZONE_0][INST_0][MESH_0]->node[iPoint]->GetDomain() && compressible){
606  Normal = geometry_container[ZONE_0][INST_0][MESH_0]->vertex[iMarker][iVertex]->GetNormal();
607  Area = 0.0;
608  for (iDim = 0; iDim < nDim; iDim++)
609  Area += Normal[iDim]*Normal[iDim];
610  Area = sqrt(Area);
611 
612  for (iDim = 0; iDim < nDim; iDim++)
613  UnitNormal[iDim] = Normal[iDim]/Area;
614 
615  laminar_viscosity = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
616  thermal_conductivity = Cp * (laminar_viscosity/Prandtl_Lam);
617  /*Compute wall heat flux (normal to the wall) based on computed temperature gradient*/
618  for(iDim=0; iDim < nDim; iDim++){
619  GradT[iDim] = solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->GetNodes()->GetGradient_Primitive(iPoint, 0, iDim);
620  dTdn += GradT[iDim]*UnitNormal[iDim];
621  }
622

624  }
