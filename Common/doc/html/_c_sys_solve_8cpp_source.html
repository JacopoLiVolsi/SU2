48 template<class ScalarType>
49 void CSysSolve<ScalarType>::GenerateGivens(ScalarType & dx, ScalarType & dy, ScalarType & s, ScalarType & c) {
50 
51  if ( (dx == 0.0) && (dy == 0.0) ) {
52  c = 1.0;
53  s = 0.0;
54  }
55  else if ( fabs(dy) > fabs(dx) ) {
56  ScalarType tmp = dx/dy;
57  dx = sqrt(1.0 + tmp*tmp);
58  s = Sign(1.0/dx, dy);
59  c = tmp*s;
60  }
61  else if ( fabs(dy) <= fabs(dx) ) {
62  ScalarType tmp = dy/dx;
63  dy = sqrt(1.0 + tmp*tmp);
64  c = Sign(1.0/dy, dx);
65  s = tmp*c;
66  }
67  else {
68  // dx and/or dy must be invalid
69  dx = 0.0;
70  dy = 0.0;
71  c = 1.0;
72  s = 0.0;
73  }
74  dx = fabs(dx*dy);
75  dy = 0.0;
76 }

93 template<class ScalarType>
94 void CSysSolve<ScalarType>::ModGramSchmidt(int i, vector<vector<ScalarType> > & Hsbg, vector<CSysVector<ScalarType> > & w) {
95 
96  bool Convergence = true;
97 
98  /*--- Parameter for reorthonormalization ---*/
99 
100  static const ScalarType reorth = 0.98;
101 
102  /*--- Get the norm of the vector being orthogonalized, and find the
103  threshold for re-orthogonalization ---*/
104 
105  ScalarType nrm = dotProd(w[i+1], w[i+1]);
106  ScalarType thr = nrm*reorth;
107 
108  /*--- The norm of w[i+1] < 0.0 or w[i+1] = NaN ---*/
109 
110  if ((nrm <= 0.0) || (nrm != nrm)) Convergence = false;
111 
112  /*--- Synchronization point to check the convergence of the solver ---*/
113 
114 #ifdef HAVE_MPI
115 
116  int rank = SU2_MPI::GetRank();
117  int size = SU2_MPI::GetSize();
118 
119  unsigned short *sbuf_conv = NULL, *rbuf_conv = NULL;
120  sbuf_conv = new unsigned short[1]; sbuf_conv[0] = 0;
121  rbuf_conv = new unsigned short[1]; rbuf_conv[0] = 0;
122 
123  /*--- Convergence criteria ---*/
124 
125  sbuf_conv[0] = Convergence;
126  SU2_MPI::Reduce(sbuf_conv, rbuf_conv, 1, MPI_UNSIGNED_SHORT, MPI_SUM, MASTER_NODE, MPI_COMM_WORLD);
127 
128  /*-- Compute global convergence criteria in the master node --*/
129 
130  sbuf_conv[0] = 0;
131  if (rank == MASTER_NODE) {
132  if (rbuf_conv[0] == size) sbuf_conv[0] = 1;
133  else sbuf_conv[0] = 0;
134  }
135 
136  SU2_MPI::Bcast(sbuf_conv, 1, MPI_UNSIGNED_SHORT, MASTER_NODE, MPI_COMM_WORLD);
137 
138  if (sbuf_conv[0] == 1) Convergence = true;
139  else Convergence = false;
140 
141  delete [] sbuf_conv;
142  delete [] rbuf_conv;
143 
144 #endif
145 
146  if (!Convergence) {
147  SU2_MPI::Error("SU2 has diverged.", CURRENT_FUNCTION);
148  }
149 
150  /*--- Begin main Gram-Schmidt loop ---*/
151 
152  for (int k = 0; k < i+1; k++) {
153  ScalarType prod = dotProd(w[i+1], w[k]);
154  Hsbg[k][i] = prod;
155  w[i+1].Plus_AX(-prod, w[k]);
156 
157  /*--- Check if reorthogonalization is necessary ---*/
158 
159  if (prod*prod > thr) {
160  prod = dotProd(w[i+1], w[k]);
161  Hsbg[k][i] += prod;
162  w[i+1].Plus_AX(-prod, w[k]);
163  }
164 
165  /*--- Update the norm and check its size ---*/
166 
167  nrm -= Hsbg[k][i]*Hsbg[k][i];
168  if (nrm < 0.0) nrm = 0.0;
169  thr = nrm*reorth;
170  }
171 
172  /*--- Test the resulting vector ---*/
173 
174  nrm = w[i+1].norm();
175  Hsbg[i+1][i] = nrm;
176 
177  /*--- Scale the resulting vector ---*/
178 
179  w[i+1] /= nrm;
180

685  }
686 
687  /*--- Set the norm to the initial initial residual value. ---*/
688 
689  norm0 = norm_r;
690 
691  /*--- Output header information including initial residual. ---*/
692 
693  if ((monitoring) && (rank == MASTER_NODE)) {
694  cout << "\n# " << "Smoother" << " residual history" << endl;
695  cout << "# Residual tolerance target = " << tol << endl;
696  cout << "# Initial residual norm = " << norm_r << endl;
697  cout << " " << i << " " << norm_r/norm0 << endl;
698  }
699 
700  }
701 
702  /*--- Smoothing Iterations ---*/
703 
704  for (i=0; i<m; i++) {
705 
706  /*--- Compute the solution increment (z), or "search" direction, by applying
707  the preconditioner to the residual, i.e. z = M^{-1} * r. ---*/
708 
709  precond(r, z);
710 
711  /*--- The increment will be added to the solution (with relaxation omega)
712  to update the residual, needed to compute the next increment, we get the
713  product of the matrix with the direction (into A_x) and subtract from the
714  current residual, the system is linear so this saves some computation
715  compared to re-evaluating r = b-A*x. ---*/
716 
717  mat_vec(z, A_x);
718 
719  /*--- Update solution and residual with relaxation omega. Mathematically this
720  is a modified Richardson iteration for the left-preconditioned system
721  M^{-1}(b-A*x) which converges if ||I-w*M^{-1}*A|| < 1. Combining this method
722  with a Gauss-Seidel preconditioner and w>1 is NOT equivalent to SOR. ---*/
723 
724  x.Plus_AX(omega, z);
725  r.Plus_AX(-omega, A_x);
726 
727  /*--- Only compute the residuals in full communication mode. ---*/
728  /*--- Check if solution has converged, else output the relative residual if necessary. ---*/
729 
730  if (config->GetComm_Level() == COMM_FULL) {
731  norm_r = r.norm();
732  if (norm_r < tol*norm0) break;
733  if (((monitoring) && (rank == MASTER_NODE)) && ((i+1) % 5 == 0))
734  cout << " " << i << " " << norm_r/norm0 << endl;
735  }
736  }
737 
738  if ((monitoring) && (rank == MASTER_NODE) && (config->GetComm_Level() == COMM_FULL)) {
739  cout << "# Smoother final (true) residual:" << endl;
740  cout << "# Iteration = " << i << ": |res|/|res0| = " << norm_r/norm0 << ".\n" << endl;
741  }
742 
743  (*residual) = norm_r/norm0;
744  return i;
745 }

792 template<class ScalarType>
793 unsigned long CSysSolve<ScalarType>::Solve(CSysMatrix<ScalarType> & Jacobian, CSysVector<su2double> & LinSysRes,
794  CSysVector<su2double> & LinSysSol, CGeometry *geometry, CConfig *config) {
795  /*---
796  A word about the templated types. It is assumed that the residual and solution vectors are always of su2doubles,
797  meaning that they are active in the discrete adjoint. The same assumption is made in SetExternalSolve.
798  When the Jacobian is passive (and therefore not compatible with the vectors) we go through the "HandleTemporaries"
799  mechanisms. Note that CG, BCGSTAB, and FGMRES, all expect the vector to be compatible with the Product and
800  Preconditioner (and therefore with the Matrix). Likewise for Solve_b (which is used by CSysSolve_b).
801  There are no provisions here for active Matrix and passive Vectors as that makes no sense since we only handle the
802  derivatives of the residual in CSysSolve_b.
803  ---*/
804 
805  unsigned short KindSolver, KindPrecond;
806  unsigned long MaxIter, RestartIter, IterLinSol = 0;
807  ScalarType SolverTol, Norm0 = 0.0;
808  bool ScreenOutput;
809 
810  /*--- Normal mode ---*/
811 
812  if(!mesh_deform) {
813 
814  KindSolver = config->GetKind_Linear_Solver();
815  KindPrecond = config->GetKind_Linear_Solver_Prec();
816  MaxIter = config->GetLinear_Solver_Iter();
817  RestartIter = config->GetLinear_Solver_Restart_Frequency();
818  SolverTol = SU2_TYPE::GetValue(config->GetLinear_Solver_Error());
819  ScreenOutput = false;
820  }
821 
822  /*--- Mesh Deformation mode ---*/
823 
824  else {
825 
826  KindSolver = config->GetKind_Deform_Linear_Solver();
827  KindPrecond = config->GetKind_Deform_Linear_Solver_Prec();
828  MaxIter = config->GetDeform_Linear_Solver_Iter();
829  RestartIter = config->GetLinear_Solver_Restart_Frequency();
830  SolverTol = SU2_TYPE::GetValue(config->GetDeform_Linear_Solver_Error());
831  ScreenOutput = config->GetDeform_Output();
832  }
833 
834  /*--- Stop the recording for the linear solver ---*/
835 
836  bool TapeActive = NO;
837 
838  if (config->GetDiscrete_Adjoint()) {
839 #ifdef CODI_REVERSE_TYPE
840 
841  TapeActive = AD::globalTape.isActive();
842 
843  AD::StartExtFunc(false, false);
844 
845  AD::SetExtFuncIn(&LinSysRes[0], LinSysRes.GetLocSize());
846 
847  AD::StopRecording();
848 #endif
849  }
850 
851  /*--- Create matrix-vector product, preconditioner, and solve the linear system ---*/
852 
853  HandleTemporariesIn(LinSysRes, LinSysSol);
854 
855  CMatrixVectorProduct<ScalarType>* mat_vec = new CSysMatrixVectorProduct<ScalarType>(Jacobian, geometry, config);
856  CPreconditioner<ScalarType>* precond = NULL;
857 
858  switch (KindPrecond) {
859  case JACOBI:
860  Jacobian.BuildJacobiPreconditioner();
861  precond = new CJacobiPreconditioner<ScalarType>(Jacobian, geometry, config);
862  break;
863  case ILU:
864  Jacobian.BuildILUPreconditioner();
865  precond = new CILUPreconditioner<ScalarType>(Jacobian, geometry, config);
866  break;
867  case LU_SGS:
868  precond = new CLU_SGSPreconditioner<ScalarType>(Jacobian, geometry, config);
869  break;
870  case LINELET:
871  Jacobian.BuildJacobiPreconditioner();
872  precond = new CLineletPreconditioner<ScalarType>(Jacobian, geometry, config);
873  break;
874  case PASTIX_ILU: case PASTIX_LU_P: case PASTIX_LDLT_P:
875  Jacobian.BuildPastixPreconditioner(geometry, config, KindPrecond);
876  precond = new CPastixPreconditioner<ScalarType>(Jacobian, geometry, config);
877  break;
878  default:
879  Jacobian.BuildJacobiPreconditioner();
880  precond = new CJacobiPreconditioner<ScalarType>(Jacobian, geometry, config);
881  break;
882  }
883 
884  switch (KindSolver) {
885  case BCGSTAB:
886  IterLinSol = BCGSTAB_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, *mat_vec, *precond, SolverTol, MaxIter, &Residual, ScreenOutput, config);
887  break;
888  case FGMRES:
889  IterLinSol = FGMRES_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, *mat_vec, *precond, SolverTol, MaxIter, &Residual, ScreenOutput, config);
890  break;
891  case CONJUGATE_GRADIENT:
892  IterLinSol = CG_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, *mat_vec, *precond, SolverTol, MaxIter, &Residual, ScreenOutput, config);
893  break;
894  case RESTARTED_FGMRES:
895  IterLinSol = 0;
896  Norm0 = LinSysRes_ptr->norm();
897  while (IterLinSol < MaxIter) {
898  /*--- Enforce a hard limit on total number of iterations ---*/
899  unsigned long IterLimit = min(RestartIter, MaxIter-IterLinSol);
900  IterLinSol += FGMRES_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, *mat_vec, *precond, SolverTol, IterLimit, &Residual, ScreenOutput, config);
901  if ( Residual < SolverTol*Norm0 ) break;
902  }
903  break;
904  case SMOOTHER:
905  IterLinSol = Smoother_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, *mat_vec, *precond, SolverTol, MaxIter, &Residual, ScreenOutput, config);
906  break;
907  case PASTIX_LDLT : case PASTIX_LU:
908  Jacobian.BuildPastixPreconditioner(geometry, config, KindSolver);
909  Jacobian.ComputePastixPreconditioner(*LinSysRes_ptr, *LinSysSol_ptr, geometry, config);
910  IterLinSol = 1;
911  break;
912  default:
913  SU2_MPI::Error("Unknown type of linear solver.",CURRENT_FUNCTION);
914  }
915 
916  delete mat_vec;
917  delete precond;
918 
919  HandleTemporariesOut(LinSysSol);
920 
921  if(TapeActive) {
922 
923  bool RequiresTranspose = !mesh_deform; // jacobian is symmetric
924  if (!mesh_deform) KindPrecond = config->GetKind_DiscAdj_Linear_Prec();
925  else KindPrecond = config->GetKind_Deform_Linear_Solver_Prec();
926 
927  /*--- Start recording if it was stopped for the linear solver ---*/
928 
929  AD::StartRecording();
930 
931  AD::SetExtFuncOut(&LinSysSol[0], (int)LinSysSol.GetLocSize());
932 
933 #ifdef CODI_REVERSE_TYPE
934  AD::FuncHelper->addUserData(&LinSysRes);
935  AD::FuncHelper->addUserData(&LinSysSol);
936  AD::FuncHelper->addUserData(&Jacobian);
937  AD::FuncHelper->addUserData(geometry);
938  AD::FuncHelper->addUserData(config);
939  AD::FuncHelper->addUserData(this);
940  AD::FuncHelper->addToTape(CSysSolve_b<ScalarType>::Solve_b);
941 #endif
942 
943  /*--- Build preconditioner for the transposed Jacobian ---*/
944 
945  switch(KindPrecond) {
946  case ILU:
947  Jacobian.BuildILUPreconditioner(RequiresTranspose);
948  break;
949  case JACOBI:
950  Jacobian.BuildJacobiPreconditioner(RequiresTranspose);
951  break;
952  case PASTIX_ILU: case PASTIX_LU_P: case PASTIX_LDLT_P:
953  Jacobian.BuildPastixPreconditioner(geometry, config, KindPrecond, RequiresTranspose);
954  break;
955  default:
956  SU2_MPI::Error("The specified preconditioner is not yet implemented for the discrete adjoint method.", CURRENT_FUNCTION);
957  break;
958  }
959 
960  AD::EndExtFunc();
961  }
962 
963  return IterLinSol;
964 }
