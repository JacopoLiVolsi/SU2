356 void CSurfaceFEMDataSorter::SortSurfaceConnectivity(CConfig *config, CGeometry *geometry, unsigned short Elem_Type) {
357 
358  /* Determine the number of nodes for this element type. */
359  unsigned short NODES_PER_ELEMENT = 0;
360  switch (Elem_Type) {
361  case LINE:
362  NODES_PER_ELEMENT = N_POINTS_LINE;
363  break;
364  case TRIANGLE:
365  NODES_PER_ELEMENT = N_POINTS_TRIANGLE;
366  break;
367  case QUADRILATERAL:
368  NODES_PER_ELEMENT = N_POINTS_QUADRILATERAL;
369  break;
370  default:
371  SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
372  }
373 
374  /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
375  geometrical information for the FEM DG solver. ---*/
376  CMeshFEM_DG *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
377 
378  unsigned long nVolElemOwned = DGGeometry->GetNVolElemOwned();
379  CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
380 
381  const CBoundaryFEM *boundaries = DGGeometry->GetBoundaries();
382  const CFEMStandardBoundaryFace *standardBoundaryFacesSol = DGGeometry->GetStandardBoundaryFacesSol();
383 
384  /*--- Create the map from the global DOF ID to the local index.
385  Note one is added to the index value, because visualization
386  softwares typically use 1-based indexing. ---*/
387  vector<unsigned long> globalID;
388  for(unsigned long l=0; l<nVolElemOwned; ++l) {
389  for(unsigned short j=0; j<volElem[l].nDOFsSol; ++j) {
390  const unsigned long globalIndex = volElem[l].offsetDOFsSolGlobal + j + 1;
391  globalID.push_back(globalIndex);
392  }
393  }
394 
395  /*--- Determine the number of sub-elements on this rank by looping
396  over the surface elements of the boundary markers that must
397  be plotted. ---*/
398  unsigned long nSubElem_Local = 0;
399  for(unsigned short iMarker=0; iMarker < config->GetnMarker_All(); ++iMarker) {
400  if( !boundaries[iMarker].periodicBoundary ) {
401  if (config->GetMarker_All_Plotting(iMarker) == YES) {
402  const vector<CSurfaceElementFEM> &surfElem = boundaries[iMarker].surfElem;
403  for(unsigned long i=0; i<surfElem.size(); ++i) {
404  const unsigned short ind = surfElem[i].indStandardElement;
405  const unsigned short VTK_Type = standardBoundaryFacesSol[ind].GetVTK_Type();
406  if(Elem_Type == VTK_Type) nSubElem_Local += standardBoundaryFacesSol[ind].GetNSubFaces();
407  }
408  }
409  }
410  }
411 
412  /* Allocate the memory to store the connectivity if the size is
413  larger than zero. */
414  int *Conn_SubElem = NULL;
415  if(nSubElem_Local > 0) Conn_SubElem = new int[nSubElem_Local*NODES_PER_ELEMENT]();
416 
417  /*--- Repeat the loop over the surface elements of the boundary markers
418  that must be plotted, but now store the connectivity. ---*/
419  unsigned long kNode = 0;
420  for(unsigned short iMarker=0; iMarker < config->GetnMarker_All(); ++iMarker) {
421  if( !boundaries[iMarker].periodicBoundary ) {
422  if (config->GetMarker_All_Plotting(iMarker) == YES) {
423  const vector<CSurfaceElementFEM> &surfElem = boundaries[iMarker].surfElem;
424 
425  /* Loop over the surface elements of this boundary marker. */
426  for(unsigned long i=0; i<surfElem.size(); ++i) {
427 
428  /* Check if this is the element type to be stored. */
429  const unsigned short ind = surfElem[i].indStandardElement;
430  const unsigned short VTK_Type = standardBoundaryFacesSol[ind].GetVTK_Type();
431  if(Elem_Type == VTK_Type) {
432 
433  /* Get the number of sub-elements and the local connectivity of
434  the sub-elements. */
435  const unsigned short nSubFaces = standardBoundaryFacesSol[ind].GetNSubFaces();
436  const unsigned short *connSubFaces = standardBoundaryFacesSol[ind].GetSubFaceConn();
437 
438  /* Store the global connectivities. */
439  const unsigned short kk = NODES_PER_ELEMENT*nSubFaces;
440  for(unsigned short k=0; k<kk; ++k, ++kNode)
441  Conn_SubElem[kNode] = globalID[surfElem[i].DOFsSolFace[connSubFaces[k]]];
442  }
443  }
444  }
445  }
446  }
447 
448  /*--- Store the particular global element count in the class data,
449  and set the class data pointer to the connectivity array. ---*/
450  switch (Elem_Type) {
451  case LINE:
452  nParallel_Line = nSubElem_Local;
453  if (Conn_Line_Par != NULL) delete [] Conn_Line_Par;
454  Conn_Line_Par = Conn_SubElem;
455  break;
456  case TRIANGLE:
457  nParallel_Tria = nSubElem_Local;
458  if (Conn_Tria_Par != NULL) delete [] Conn_Tria_Par;
459  Conn_Tria_Par = Conn_SubElem;
460  break;
461  case QUADRILATERAL:
462  nParallel_Quad = nSubElem_Local;
463  if (Conn_Quad_Par != NULL) delete [] Conn_Quad_Par;
464  Conn_Quad_Par = Conn_SubElem;
465  break;
466  default:
467  SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
468  break;
469  }
