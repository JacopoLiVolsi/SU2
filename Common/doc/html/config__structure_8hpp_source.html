1163  // Check if the key is already in the map. If this fails, it is coder error
1164  // and not user error, so throw.
1165  assert(option_map.find(name) == option_map.end());
1166 
1167  // Add this option to the list of all the options
1168  all_options.insert(pair<string, bool>(name, true));
1169 
1170  // Create the parser for a su2double option with a reference to the option_field and the desired
1171  // default value. This will take the string in the config file, convert it to a su2double, and
1172  // place that su2double in the memory location specified by the reference.
1173  COptionBase* val = new COptionDouble(name, option_field, default_value);
1174 
1175  // Create an association between the option name ("CFL") and the parser generated above.
1176  // During configuration, the parsing script will get the option name, and use this map
1177  // to find how to parse that option.
1178  option_map.insert(pair<string, COptionBase *>(name, val));
1179  }
