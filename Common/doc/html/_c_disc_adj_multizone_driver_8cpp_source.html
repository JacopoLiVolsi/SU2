171  * correctly as the OF gradient will overwrite the solution. ---*/
172

175 
176  /*--- Evaluate the objective function gradient w.r.t. the solutions of all zones. ---*/
177 
178  SetRecording(NONE, Kind_Tape::FULL_TAPE, ZONE_0);
179  SetRecording(FLOW_CONS_VARS, Kind_Tape::OBJECTIVE_FUNCTION_TAPE, ZONE_0);

185 
186  /*--- Initialize External with the objective function gradient. ---*/
187 
188  for (iZone = 0; iZone < nZone; iZone++) {
189 
190  iteration_container[iZone][INST_0]->Iterate(output_container[iZone], integration_container, geometry_container,
191  solver_container, numerics_container, config_container,
192  surface_movement, grid_movement, FFDBox, iZone, INST_0);
193  Add_Solution_To_External(iZone);
194  }
195 
196  /*--- Loop over the number of outer iterations. ---*/
197 
198  for (unsigned long iOuterIter = 0, StopCalc = false; !StopCalc; iOuterIter++) {
199 
200  driver_config->SetOuterIter(iOuterIter);
201 
202  for (iZone = 0; iZone < nZone; iZone++)
203  config_container[iZone]->SetOuterIter(iOuterIter);
204 
205  /*--- For the adjoint iteration we need the derivatives of the iteration function with
206  * respect to the state (and possibly the mesh coordinate) variables.
207  * Since these derivatives do not change in the steady state case we only have to record
208  * if the current recording is different from them.
209  *
210  * To set the tape appropriately, the following recording methods are provided:
211  * (1) NONE: All information from a previous recording is removed.
212  * (2) FLOW_CONS_VARS: State variables of all solvers in a zone as input.
213  * (3) MESH_COORDS: Mesh coordinates as input.
214  * (4) COMBINED: Mesh coordinates and state variables as input.
215  *
216  * By default, all (state and mesh coordinate variables) will be declared as output,
217  * since it does not change the computational effort. ---*/
218 
219 
220  /*--- If we want to set up zone-specific tapes (retape), we do not need to record
221  * here. Otherwise, the whole tape of a coupled run will be created. ---*/
222 
223  if (!retape && (RecordingState != FLOW_CONS_VARS)) {
224  SetRecording(NONE, Kind_Tape::FULL_TAPE, ZONE_0);
225  SetRecording(FLOW_CONS_VARS, Kind_Tape::FULL_TAPE, ZONE_0);
226  }
227

229 
230  for (iZone = 0; iZone < nZone; iZone++) {
231 
232  if (retape) {

235  }
236

238 
239  Set_Solution_To_BGSSolution_k(iZone);
240 
241  /*--- Inner loop to allow for multiple adjoint updates with respect to solvers in iZone. ---*/
242 
243  bool eval_transfer = false;

245 
246  for (unsigned long iInnerIter = 0; iInnerIter < nInnerIter[iZone]; iInnerIter++) {
247 
248  config_container[iZone]->SetInnerIter(iInnerIter);
249 
250  /*--- Add off-diagonal contribution (including the OF gradient) to Solution. ---*/
251

253  Add_External_To_Solution(iZone);
254  }
255  else {
256  /*--- If we restarted, Solution already has all contribution,
257  * we run only one inner iter to compute the cross terms. ---*/
258  eval_transfer = true;
259  }
260 
261  /*--- Evaluate the tape section belonging to solvers in iZone.
262  * Only evaluate TRANSFER terms on the last iteration or after convergence. ---*/
263 
264  eval_transfer = eval_transfer || (iInnerIter == nInnerIter[iZone]-1);
265 
266  ComputeAdjoints(iZone, eval_transfer);
267

269

271  solver_container, numerics_container, config_container,
272  surface_movement, grid_movement, FFDBox, iZone, INST_0);
273 
274  /*--- Print out the convergence data to screen and history file ---*/
275 
276  bool converged = iteration_container[iZone][INST_0]->Monitor(output_container[iZone], integration_container,
277  geometry_container, solver_container, numerics_container,
278  config_container, surface_movement, grid_movement, FFDBox, iZone, INST_0);
279  if (eval_transfer) break;
280  eval_transfer = converged;
281 
282  }
283 
284  /*--- Off-diagonal (coupling term) BGS update. ---*/
285 
286  for (unsigned short jZone = 0; jZone < nZone; jZone++) {
287 
288  if (jZone != iZone && interface_container[jZone][iZone] != NULL) {
289

291  * that is, for the cases iZone != jZone we are evaluating cross derivatives between zones. ---*/
292 
293  iteration_container[jZone][INST_0]->Iterate(output_container[jZone], integration_container, geometry_container,
294  solver_container, numerics_container, config_container,
295  surface_movement, grid_movement, FFDBox, jZone, INST_0);
296 
297  /*--- Extract the cross-term performing a relaxed update of it and of the sum (External) for jZone. ---*/
298 
299  Update_Cross_Term(iZone, jZone);
300  }
301  }
302 
303  /*--- Save Solution to Solution_BGS and compute residual from Solution_BGS and Solution_BGS_k. ---*/
304 
305  SetResidual_BGS(iZone);
306 
307  /*--- Save Solution to Solution_BGS_k for a next outer iteration.
308  * (Solution might be overwritten when entering another zone because of cross derivatives.) ---*/
309 
310  Set_BGSSolution(iZone);
311  }
312 
313  /*--- Set the multizone output. ---*/
314 
315  driver_output->SetMultizoneHistory_Output(output_container, config_container, driver_config, 0, iOuterIter);
316 
317  /*--- Check for convergence. ---*/
318 
319  StopCalc = driver_output->GetConvergence() || (iOuterIter == nOuterIter-1);
320 
321  /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
322 
323  AD::ClearAdjoints();
324

326

334 void CDiscAdjMultizoneDriver::EvaluateSensitivities(unsigned long iOuterIter, bool StopCalc) {
335 
336  /*--- SetRecording stores the computational graph on one iteration of the direct problem. Calling it with NONE
337  * as argument ensures that all information from a previous recording is removed. ---*/
338 
339  SetRecording(NONE, Kind_Tape::FULL_TAPE, ZONE_0);
340 
341  /*--- Store the computational graph of one direct iteration with the mesh coordinates as input. ---*/
342 
343  SetRecording(MESH_COORDS, Kind_Tape::FULL_TAPE, ZONE_0);
344 
345  /*--- Initialize the adjoint of the output variables of the iteration with the adjoint solution
346  * of the current iteration. The values are passed to the AD tool. ---*/
347 
348  for (iZone = 0; iZone < nZone; iZone++) {
349 
350  Set_Solution_To_BGSSolution_k(iZone);
351 
352  Add_External_To_Solution(iZone);
353 
354  iteration_container[iZone][INST_0]->InitializeAdjoint(solver_container, geometry_container,
355  config_container, iZone, INST_0);
356  }
357 
358  /*--- Initialize the adjoint of the objective function with 1.0. ---*/
359 
360  SetAdj_ObjFunction();
361 
362  /*--- Interpret the stored information by calling the corresponding routine of the AD tool. ---*/
363 
364  AD::ComputeAdjoint();
365 
366  /*--- Extract the computed sensitivity values. ---*/
367 
368  for (iZone = 0; iZone < nZone; iZone++) {
369 
370  auto config = config_container[iZone];
371  auto solvers = solver_container[iZone][INST_0][MESH_0];
372  auto geometry = geometry_container[iZone][INST_0][MESH_0];
373

377  case DISC_ADJ_INC_EULER: case DISC_ADJ_INC_NAVIER_STOKES: case DISC_ADJ_INC_RANS:
378

380  solvers[ADJMESH_SOL]->SetSensitivity(geometry, solvers, config);
381  } else {
382  solvers[ADJFLOW_SOL]->SetSensitivity(geometry, solvers, config);
383  }
384  break;
385 
386  case DISC_ADJ_HEAT:
387 
388  solvers[ADJHEAT_SOL]->SetSensitivity(geometry, solvers, config);
389  break;
390 
391  case DISC_ADJ_FEM:
392 
393  solvers[ADJFEA_SOL]->SetSensitivity(geometry, solvers, config);
394  break;
395 
396  default:
397  if (rank == MASTER_NODE)
398  cout << "WARNING: Sensitivities not set for one of the specified discrete adjoint solvers!" << endl;
399  break;
400  }
401  }
402

404 
405  AD::ClearAdjoints();
406 
407  for (iZone = 0; iZone < nZone; iZone++) {
408 
409  output_container[iZone]->SetResult_Files(geometry_container[iZone][INST_0][MESH_0],
410  config_container[iZone],
411  solver_container[iZone][INST_0][MESH_0], iOuterIter, StopCalc);
412  }

845  unsigned short nVar = solver->GetnVar();

855 void CDiscAdjMultizoneDriver::HandleDataTransfer() {
856 
857  unsigned long ExtIter = 0;
858 
859  for(iZone = 0; iZone < nZone; iZone++) {
860 
861  /*--- In principle, the mesh does not need to be updated ---*/
862  bool DeformMesh = false;
863 
864  /*--- Transfer from all the remaining zones ---*/
865  for (unsigned short jZone = 0; jZone < nZone; jZone++){
866  /*--- The target zone is iZone ---*/
867  if (jZone != iZone && interface_container[iZone][jZone] != NULL) {
868  DeformMesh = DeformMesh || Transfer_Data(jZone, iZone);
869  }
870  }
871  /*--- If a mesh update is required due to the transfer of data ---*/
872  if (DeformMesh) DynamicMeshUpdate(iZone, ExtIter);
