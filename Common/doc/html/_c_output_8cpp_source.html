657 bool COutput::Convergence_Monitoring(CConfig *config, unsigned long Iteration) {
658 
659  unsigned short iCounter;
660 
661  convergence = true;
662 
663  for (unsigned short iField_Conv = 0; iField_Conv < convFields.size(); iField_Conv++){
664 
665  bool fieldConverged = false;
666 
667  const string &convField = convFields[iField_Conv];
668  if (historyOutput_Map.count(convField) > 0){
669  su2double monitor = historyOutput_Map[convField].value;
670 
671  /*--- Cauchy based convergence criteria ---*/
672 
673  if (historyOutput_Map[convField].fieldType == HistoryFieldType::COEFFICIENT) {
674 
675  if (Iteration == 0){
676  for (iCounter = 0; iCounter < nCauchy_Elems; iCounter++){
677  cauchySerie[iField_Conv][iCounter] = 0.0;
678  }
679  newFunc[iField_Conv] = monitor;
680  }
681 
682  oldFunc[iField_Conv] = newFunc[iField_Conv];
683  newFunc[iField_Conv] = monitor;
684  cauchyFunc = fabs(newFunc[iField_Conv] - oldFunc[iField_Conv])/fabs(monitor);
685 
686  cauchySerie[iField_Conv][Iteration % nCauchy_Elems] = cauchyFunc;
687  cauchyValue = 0.0;
688  for (iCounter = 0; iCounter < nCauchy_Elems; iCounter++)
689  cauchyValue += cauchySerie[iField_Conv][iCounter];
690 
691  cauchyValue /= nCauchy_Elems;
692 
693  if (cauchyValue >= cauchyEps) { fieldConverged = false;}
694  else { fieldConverged = true;}
695 
696  /*--- Start monitoring only if the current iteration
697  * is larger than the number of cauchy elements and
698  * the number of start-up iterations --- */
699 
700  if (Iteration < max(config->GetStartConv_Iter(), nCauchy_Elems)){
701  fieldConverged = false;
702  }
703 
704  SetHistoryOutputValue("CAUCHY_" + convField, cauchyValue);
705 
706  }
707 
708 
709  /*--- Residual based convergence criteria ---*/
710 
711  if (historyOutput_Map[convField].fieldType == HistoryFieldType::RESIDUAL ||
712  historyOutput_Map[convField].fieldType == HistoryFieldType::AUTO_RESIDUAL) {
713 
714  /*--- Check the convergence ---*/
715 
716  if (Iteration != 0 && (monitor <= minLogResidual)) { fieldConverged = true; }
717  else { fieldConverged = false; }
718 
719  }
720 
721  /*--- Do not apply any convergence criteria of the number
722  of iterations is less than a particular value ---*/
723 
724  if (Iteration < config->GetStartConv_Iter()) {
725  fieldConverged = false;
726  }
727 
728  convergence = fieldConverged && convergence;
729  }
730  }
731  if (convFields.empty()) convergence = false;
732 
733  /*--- Apply the same convergence criteria to all the processors ---*/
734 
735 #ifdef HAVE_MPI
736 
737  unsigned short *sbuf_conv = NULL, *rbuf_conv = NULL;
738  sbuf_conv = new unsigned short[1]; sbuf_conv[0] = 0;
739  rbuf_conv = new unsigned short[1]; rbuf_conv[0] = 0;
740 
741  /*--- Convergence criteria ---*/
742 
743  sbuf_conv[0] = convergence;
744  SU2_MPI::Reduce(sbuf_conv, rbuf_conv, 1, MPI_UNSIGNED_SHORT, MPI_SUM, MASTER_NODE, MPI_COMM_WORLD);
745 
746  /*-- Compute global convergence criteria in the master node --*/
747 
748  sbuf_conv[0] = 0;
749  if (rank == MASTER_NODE) {
750  if (rbuf_conv[0] == size) sbuf_conv[0] = 1;
751  else sbuf_conv[0] = 0;
