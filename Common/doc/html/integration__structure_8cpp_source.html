208  bool linear_analysis = (config->GetGeometricConditions() == SMALL_DEFORMATIONS); // Linear analysis.
209  bool first_iter = (config->GetInnerIter() == 0); // Checks if it is the first iteration
210  unsigned short IterativeScheme = config->GetKind_SpaceIteScheme_FEA(); // Iterative schemes: NEWTON_RAPHSON, MODIFIED_NEWTON_RAPHSON
211  unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
212 
213  bool restart = config->GetRestart(); // Restart solution
214  bool initial_calc_restart = (SU2_TYPE::Int(config->GetTimeIter()) == SU2_TYPE::Int(config->GetRestart_Iter())); // Restart iteration
215 
216  /*--- Compute Mass Matrix ---*/
217  /*--- The mass matrix is computed only once, at the beginning of the calculation, no matter whether the ---*/
218  /*--- problem is linear or nonlinear. This is done in the preprocessing step. ---*/
219 
220  /*--- If the analysis is linear, only a the constitutive term of the stiffness matrix has to be computed ---*/
221  /*--- This is done only once, at the beginning of the calculation. From then on, K is constant ---*/
222  if ((linear_analysis && (initial_calc && first_iter)) ||
223  (linear_analysis && restart && initial_calc_restart)) {
224  solver_container[MainSolver]->Compute_StiffMatrix(geometry, numerics, config);
225  }
226  else if (!linear_analysis) {
227  /*--- If the analysis is nonlinear, also the stress terms need to be computed ---*/
228  /*--- If the method is full Newton-Raphson, the stiffness matrix and the nodal term are updated every time ---*/
229  /*--- They are calculated together to avoid looping twice over the elements ---*/
230  if (IterativeScheme == NEWTON_RAPHSON) {
231  /*--- The Jacobian is reinitialized every time in Preprocessing (before calling Space_Integration_FEM) */
232  solver_container[MainSolver]->Compute_StiffMatrix_NodalStressRes(geometry, numerics, config);
233  }
234 
235  /*--- If the method is modified Newton-Raphson, the stiffness matrix is only computed once at the beginning of the time-step ---*/
236  /*--- Nevertheless, the Nodal Stress Term has to be computed for each iteration ---*/
237  else if (IterativeScheme == MODIFIED_NEWTON_RAPHSON) {
238 
239  if (first_iter) {
240  solver_container[MainSolver]->Compute_StiffMatrix_NodalStressRes(geometry, numerics, config);
241  }
242 
243  else {
244  solver_container[MainSolver]->Compute_NodalStressRes(geometry, numerics, config);
245  }
246 
247  }
248 
249  }
250 
251  /*--- Apply the NATURAL BOUNDARY CONDITIONS (loads). ---*/
252  /*--- If there are FSI loads, they have to be previously applied at other level involving both zones. ---*/
253 
254  /*--- Some external loads may be considered constant over the time step ---*/
255  if (first_iter) {
256  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
257  switch (config->GetMarker_All_KindBC(iMarker)) {
258  case LOAD_DIR_BOUNDARY:
259  solver_container[MainSolver]->BC_Dir_Load(geometry, numerics[FEA_TERM], config, iMarker);
260  break;
261  case LOAD_SINE_BOUNDARY:
262  solver_container[MainSolver]->BC_Sine_Load(geometry, numerics[FEA_TERM], config, iMarker);
263  break;
264  }
265  }
266  }
267 
268  /*--- Others are not, because they depend on the geometry ---*/
269  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
270  switch (config->GetMarker_All_KindBC(iMarker)) {
271  case LOAD_BOUNDARY:
272  solver_container[MainSolver]->BC_Normal_Load(geometry, numerics[FEA_TERM], config, iMarker);
273  break;
274  case DAMPER_BOUNDARY:
275  solver_container[MainSolver]->BC_Damper(geometry, numerics[FEA_TERM], config, iMarker);
