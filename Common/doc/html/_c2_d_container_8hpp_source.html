374  size_t m_resize(Index_t rows, Index_t cols) noexcept
375  {
376  /*--- fully static, no allocation needed ---*/
377  if(StaticRows!=DynamicSize && StaticCols!=DynamicSize)
378  return StaticRows*StaticCols;
379 
380  /*--- dynamic row vector, swap size specification ---*/
381  if(StaticRows==1 && StaticCols==DynamicSize) {cols = rows; rows = 1;}
382 
383  /*--- assert a static size is not being asked to change ---*/
384  if(StaticRows!=DynamicSize) assert(rows==StaticRows && "A static size was asked to change.");
385  if(StaticCols!=DynamicSize) assert(cols==StaticCols && "A static size was asked to change.");
386 
387  /*--- "rectify" sizes before continuing as asserts are usually dissabled ---*/
388  rows = (StaticRows!=DynamicSize)? StaticRows : rows;
389  cols = (StaticCols!=DynamicSize)? StaticCols : cols;
390 
391  /*--- number of requested elements ---*/
392  size_t reqSize = rows*cols;
393 
394  /*--- compare with current dimensions to determine if deallocation
395  is needed, also makes the container safe against self assignment
396  no need to check for 0 size as the allocators handle that ---*/
397  if(m_data!=nullptr)
398  {
399  if(rows==this->rows() && cols==this->cols())
400  return reqSize;
401  free(m_data);
402  }
403 
404  /*--- round up size to a multiple of the alignment specification if necessary ---*/
405  size_t bytes = reqSize*sizeof(Scalar_t);
406  size_t allocSize = (AlignSize==0)? bytes : ((bytes+AlignSize-1)/AlignSize)*AlignSize;
407 
408  /*--- request actual allocation to base class as it needs specialization ---*/
409  m_allocate(allocSize,rows,cols);
410 
411  return reqSize;
412  }
